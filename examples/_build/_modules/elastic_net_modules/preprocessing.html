"""
Preprocessing Module for Elastic Net Analysis

This module handles all data preparation steps before Elastic Net modeling:
1. DNA content labeling (manual threshold or GMM: 2N vs not_2N)
2. Proliferation labeling (manual threshold or GMM: prolif vs nonprolif)
3. Optional percentile-based filtering (e.g., Fra1 low cells)
4. Ploidy filtering (optional 2N selection)
5. Feature extraction

Author: Stallaert Lab
Date: November 13, 2025
"""

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from pathlib import Path
from typing import Optional, Tuple, Dict, List, Union, TYPE_CHECKING

# Import cc-mapping thresholding module
from cc_mapping.thresholding import GMMThresholding

if TYPE_CHECKING:
    from .results_manager import ElasticNetResultsManager


def _apply_manual_threshold_labeling(
    adata,
    feature: str,
    threshold: float,
    ordered_labels: List[str],
    label_obs_save_str: str,
    cell_line: Optional[str] = None,
    output_dir: Optional[Path] = None,
    save_histogram: bool = True,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    plot_title: Optional[str] = None
):
    """
    Apply manual threshold labeling using cc-mapping.
    
    Parameters
    ----------
    adata : AnnData
        Input data
    feature : str
        Feature name to threshold
    threshold : float
        Threshold value (values < threshold get first label, >= get second label)
    ordered_labels : list of str
        Labels for each bin (e.g., ['2N', 'not_2N'])
    label_obs_save_str : str
        Column name to save labels in adata.obs
    cell_line : str, optional
        Cell line name for plot title
    output_dir : Path, optional
        Directory to save plots
    save_histogram : bool
        Whether to save histogram
    results_manager : ElasticNetResultsManager, optional
        Results manager for saving plots
    plot_title : str, optional
        Custom plot title
        
    Returns
    -------
    adata : AnnData
        Updated adata with new label column
    """
    print(f"\n{'='*80}")
    print(f"Manual Threshold Labeling: {feature}")
    print(f"{'='*80}")
    print(f"Threshold: {threshold}")
    print(f"Labels: {ordered_labels}")
    
    # Initialize GMMThresholding object (for plotting and labeling utilities)
    gmm_obj = GMMThresholding(
        adata=adata,
        feature=feature,
        label_obs_save_str=label_obs_save_str,
        thresholding_events_key='manual_thresholding_events',
    )
    
    # Apply manual threshold directly (no fit() needed!)
    gmm_obj.categorize_samples(
        manual_thresholds=[threshold],
        ordered_labels=ordered_labels
    )
    
    # Get updated adata
    adata = gmm_obj.return_adata()
    
    # Create numeric labels for backward compatibility
    # Assuming first label is 0, second is 1
    adata.obs[label_obs_save_str.replace('_str', '')] = (
        adata.obs[label_obs_save_str] == ordered_labels[1]
    ).astype(int)
    
    # Print summary
    label_counts = adata.obs[label_obs_save_str].value_counts()
    total = len(adata)
    print(f"\nTotal cells: {total:,}")
    for label, count in label_counts.items():
        print(f"{label}: {count:,} ({count/total*100:.1f}%)")
    
    # Save histogram
    if save_histogram and output_dir is not None:
        if plot_title is None:
            plot_title = f'Manual Threshold - {cell_line}' if cell_line else 'Manual Threshold'
        
        fig = plt.figure(figsize=(10, 6))
        ax = gmm_obj.plot_hist_distribution_with_boundaries(
            title=plot_title,
            ax=plt.gca()
        )
        
        # Save using manager if available, else save to output_dir
        # Use consistent naming: ploidy/proliferation + manual/gmm
        if 'ploidy' in label_obs_save_str:
            plot_filename = 'ploidy_manual_histogram'
        elif 'prolif' in label_obs_save_str:
            plot_filename = 'proliferation_manual_histogram'
        else:
            plot_filename = label_obs_save_str.replace('_label_str', '_manual_histogram')
        
        if results_manager is not None:
            results_manager.save_preprocessing_plot(fig, plot_filename)
            print(f"Saved histogram via results_manager")
        else:
            save_path = output_dir / f'{plot_filename}.png'
            fig.savefig(save_path, dpi=150, bbox_inches='tight')
            print(f"Saved histogram: {save_path}")
        
        plt.close(fig)
    
    print(f"{'='*80}\n")
    return adata


def _apply_gmm_threshold_labeling(
    adata,
    feature: str,
    n_components: int,
    ordered_labels: List[str],
    label_obs_save_str: str,
    gmm_kwargs: Optional[Dict] = None,
    cell_line: Optional[str] = None,
    output_dir: Optional[Path] = None,
    save_histogram: bool = True,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    plot_title: Optional[str] = None
):
    """
    Apply GMM-based threshold labeling using cc-mapping.
    
    Parameters
    ----------
    adata : AnnData
        Input data
    feature : str
        Feature name to threshold
    n_components : int
        Number of GMM components
    ordered_labels : list of str
        Labels for each component (low to high)
    label_obs_save_str : str
        Column name to save labels in adata.obs
    gmm_kwargs : dict, optional
        Keyword arguments for GaussianMixture
    cell_line : str, optional
        Cell line name for plot title
    output_dir : Path, optional
        Directory to save plots
    save_histogram : bool
        Whether to save histogram
    results_manager : ElasticNetResultsManager, optional
        Results manager for saving plots
    plot_title : str, optional
        Custom plot title
        
    Returns
    -------
    adata : AnnData
        Updated adata with new label column
    """
    print(f"\n{'='*80}")
    print(f"GMM Labeling: {feature}")
    print(f"{'='*80}")
    print(f"GMM components: {n_components}")
    print(f"Ordered labels: {ordered_labels}")
    
    # Set default GMM kwargs
    if gmm_kwargs is None:
        gmm_kwargs = {
            'init_params': 'k-means++',
            'n_init': 10,
            'max_iter': 1000,
            'random_state': 42
        }
    
    # Initialize GMM thresholding
    gmm_obj = GMMThresholding(
        adata=adata,
        feature=feature,
        label_obs_save_str=label_obs_save_str,
        thresholding_events_key='gmm_thresholding_events',
        gmm_kwargs=gmm_kwargs,
    )
    
    # Fit GMM
    gmm_obj.fit(n_components=n_components)
    
    # Categorize samples
    gmm_obj.categorize_samples(
        ordered_labels=ordered_labels,
        duplicate_labels=True  # Allow collapsing multiple components
    )
    
    # Get updated adata
    adata = gmm_obj.return_adata()
    
    # Create numeric labels for backward compatibility
    # Get unique labels to determine mapping
    unique_labels = list(dict.fromkeys(ordered_labels))
    # Assuming last unique label is "positive" class (1)
    adata.obs[label_obs_save_str.replace('_str', '')] = (
        adata.obs[label_obs_save_str] == unique_labels[-1]
    ).astype(int)
    
    # Print summary
    label_counts = adata.obs[label_obs_save_str].value_counts()
    total = len(adata)
    print(f"\nTotal cells: {total:,}")
    for label, count in label_counts.items():
        print(f"{label}: {count:,} ({count/total*100:.1f}%)")
    
    # Save histogram
    if save_histogram and output_dir is not None:
        if plot_title is None:
            plot_title = f'GMM - {cell_line}' if cell_line else 'GMM'
        
        fig = plt.figure(figsize=(10, 6))
        ax = gmm_obj.plot_hist_distribution_with_boundaries(
            title=plot_title,
            ax=plt.gca()
        )
        
        # Save using manager if available, else save to output_dir
        # Use consistent naming: ploidy/proliferation + manual/gmm
        if 'ploidy' in label_obs_save_str:
            plot_filename = 'ploidy_gmm_histogram'
        elif 'prolif' in label_obs_save_str:
            plot_filename = 'proliferation_gmm_histogram'
        else:
            plot_filename = label_obs_save_str.replace('_label_str', '_gmm_histogram')
        
        if results_manager is not None:
            results_manager.save_preprocessing_plot(fig, plot_filename)
            print(f"Saved histogram via results_manager")
        else:
            save_path = output_dir / f'{plot_filename}.png'
            fig.savefig(save_path, dpi=150, bbox_inches='tight')
            print(f"Saved histogram: {save_path}")
        
        plt.close(fig)
    
    print(f"{'='*80}\n")
    return adata

def label_ploidy_and_proliferation(
    adata,
    dna_feature: str = 'norm_Int_Intg_DNA_nuc',
    prolif_marker: Tuple[str, str] = ('Int_Med_pRB_nuc', 'Int_Med_RB_nuc'),
    ploidy_config: Optional[Dict] = None,
    prolif_config: Optional[Dict] = None,
    gmm_kwargs: Optional[Dict] = None,
    output_dir: Optional[Path] = None,
    cell_line: Optional[str] = None,
    save_histograms: bool = True,
    results_manager: Optional['ElasticNetResultsManager'] = None
):
    """
    Label ALL cells for ploidy and proliferation status.
    
    Supports both manual thresholds and GMM-based labeling using cc-mapping.
    
    Parameters
    ----------
    adata : AnnData
        Single-cell data object with raw measurements
    dna_feature : str
        Feature name for DNA content
    prolif_marker : tuple of str
        (numerator, denominator) for proliferation ratio
    ploidy_config : dict or None
        Configuration for ploidy labeling:
        {
            'enabled': bool (default True),
            'use_manual': bool (default False),
            'manual_threshold': float (if use_manual=True),
            'manual_labels': list (if use_manual=True),
            'n_components': int (if use_manual=False),
            'ordered_labels': list (if use_manual=False),
            'save_histogram': bool (default True)
        }
    prolif_config : dict or None
        Configuration for proliferation labeling:
        {
            'use_manual': bool (default False),
            'manual_threshold': float (if use_manual=True),
            'manual_labels': list (if use_manual=True),
            'n_components': int (if use_manual=False),
            'ordered_labels': list (if use_manual=False),
            'save_histogram': bool (default True)
        }
    gmm_kwargs : dict or None
        Keyword arguments for GaussianMixture (only used for GMM mode)
    output_dir : Path or None
        Directory to save histograms
    cell_line : str or None
        Cell line name for plot titles
    save_histograms : bool
        Master switch for saving histograms
    results_manager : ElasticNetResultsManager or None
        If provided, saves histograms through manager
        
    Returns
    -------
    adata : AnnData
        Input adata with new columns in .obs:
        - 'ploidy_label': int (0=not_2N, 1=2N)
        - 'ploidy_label_str': str ('not_2N', '2N')
        - 'prolif_label': int (0=nonprolif, 1=prolif)
        - 'prolif_label_str': str ('nonprolif', 'prolif')
        - 'prolif_marker_log': float (log10 pRB/RB ratio)
    """
    # Set defaults
    if ploidy_config is None:
        ploidy_config = {
            'enabled': True,
            'use_manual': False,
            'n_components': 3,
            'ordered_labels': ['2N', 'not_2N', 'not_2N'],
            'save_histogram': True
        }
    
    if prolif_config is None:
        prolif_config = {
            'use_manual': False,
            'n_components': 2,
            'ordered_labels': ['nonprolif', 'prolif'],
            'save_histogram': True
        }
    
    # Set default GMM kwargs if not provided
    if gmm_kwargs is None:
        gmm_kwargs = {
            'init_params': 'k-means++',
            'n_init': 10,
            'max_iter': 1000,
            'random_state': 42
        }
    
    # Determine output directory
    if save_histograms:
        if results_manager is not None:
            output_dir = Path(results_manager.dirs['preprocessing'])
        elif output_dir is not None:
            output_dir = Path(output_dir)
            output_dir.mkdir(parents=True, exist_ok=True)
    
    # ========================================================================
    # Step 1: Ploidy Labeling (Manual or GMM)
    # ========================================================================
    if ploidy_config.get('enabled', True):
        use_manual = ploidy_config.get('use_manual', False)
        
        if use_manual:
            # Manual threshold mode
            threshold = ploidy_config.get('manual_threshold')
            labels = ploidy_config.get('manual_labels', ['2N', 'not_2N'])
            
            if threshold is None:
                raise ValueError("ploidy_config['manual_threshold'] must be provided when use_manual=True")
            
            adata = _apply_manual_threshold_labeling(
                adata=adata,
                feature=dna_feature,
                threshold=threshold,
                ordered_labels=labels,
                label_obs_save_str='ploidy_label_str',
                cell_line=cell_line,
                output_dir=output_dir,
                save_histogram=ploidy_config.get('save_histogram', True) and save_histograms,
                results_manager=results_manager,
                plot_title=f'DNA Content (Manual) - {cell_line}' if cell_line else 'DNA Content (Manual)'
            )
        else:
            # GMM mode
            n_components = ploidy_config.get('n_components', 3)
            ordered_labels = ploidy_config.get('ordered_labels')
            
            if ordered_labels is None:
                raise ValueError(
                    "ploidy_config['ordered_labels'] must be provided for GMM mode. "
                    f"Example for {n_components} components: ['2N', 'not_2N', 'not_2N']"
                )
            
            adata = _apply_gmm_threshold_labeling(
                adata=adata,
                feature=dna_feature,
                n_components=n_components,
                ordered_labels=ordered_labels,
                label_obs_save_str='ploidy_label_str',
                gmm_kwargs=gmm_kwargs,
                cell_line=cell_line,
                output_dir=output_dir,
                save_histogram=ploidy_config.get('save_histogram', True) and save_histograms,
                results_manager=results_manager,
                plot_title=f'DNA Content GMM - {cell_line}' if cell_line else 'DNA Content GMM'
            )
    
    # ========================================================================
    # Step 2: Proliferation Labeling (Manual or GMM)
    # ========================================================================
    # Calculate proliferation ratio: pRB/RB
    numerator = adata.obs_vector(prolif_marker[0])
    denominator = adata.obs_vector(prolif_marker[1])
    # Add small epsilon only to denominator to avoid division by zero
    ratio = numerator / (denominator + 1e-10)
    # Data should already be filtered; minimal pseudocount for safety
    prolif_log = np.log10(ratio + 1e-100)
    
    # Add to obs for reference
    adata.obs['prolif_marker_log'] = prolif_log
    
    use_manual = prolif_config.get('use_manual', False)
    
    if use_manual:
        # Manual threshold mode
        # Create temporary AnnData with just the composite feature
        import anndata as ad
        temp_feature_name = 'prolif_ratio_temp'
        adata_temp = ad.AnnData(
            X=prolif_log.reshape(-1, 1),
            obs=adata.obs.copy(),
        )
        adata_temp.var_names = [temp_feature_name]
        
        threshold = prolif_config.get('manual_threshold')
        labels = prolif_config.get('manual_labels', ['nonprolif', 'prolif'])
        
        if threshold is None:
            raise ValueError("prolif_config['manual_threshold'] must be provided when use_manual=True")
        
        adata_temp = _apply_manual_threshold_labeling(
            adata=adata_temp,
            feature=temp_feature_name,
            threshold=threshold,
            ordered_labels=labels,
            label_obs_save_str='prolif_label_str',
            cell_line=cell_line,
            output_dir=output_dir,
            save_histogram=prolif_config.get('save_histogram', True) and save_histograms,
            results_manager=results_manager,
            plot_title=f'Proliferation (Manual) - {cell_line}' if cell_line else 'Proliferation (Manual)'
        )
        
        # Copy labels back to original adata
        adata.obs['prolif_label_str'] = adata_temp.obs['prolif_label_str']
        adata.obs['prolif_label'] = adata_temp.obs['prolif_label']
        
    else:
        # GMM mode
        # Create temporary AnnData with just the composite feature
        import anndata as ad
        temp_feature_name = 'prolif_ratio_temp'
        adata_temp = ad.AnnData(
            X=prolif_log.reshape(-1, 1),
            obs=adata.obs.copy(),
        )
        adata_temp.var_names = [temp_feature_name]
        
        n_components = prolif_config.get('n_components', 2)
        ordered_labels = prolif_config.get('ordered_labels')
        
        if ordered_labels is None:
            raise ValueError(
                "prolif_config['ordered_labels'] must be provided for GMM mode. "
                f"Example for {n_components} components: "
                f"2-comp: ['nonprolif', 'prolif'], "
                f"3-comp: ['nonprolif', 'nonprolif', 'prolif']"
            )
        
        adata_temp = _apply_gmm_threshold_labeling(
            adata=adata_temp,
            feature=temp_feature_name,
            n_components=n_components,
            ordered_labels=ordered_labels,
            label_obs_save_str='prolif_label_str',
            gmm_kwargs=gmm_kwargs,
            cell_line=cell_line,
            output_dir=output_dir,
            save_histogram=prolif_config.get('save_histogram', True) and save_histograms,
            results_manager=results_manager,
            plot_title=f'Proliferation GMM - {cell_line}' if cell_line else 'Proliferation GMM'
        )
        
        # Copy labels back to original adata
        adata.obs['prolif_label_str'] = adata_temp.obs['prolif_label_str']
        adata.obs['prolif_label'] = adata_temp.obs['prolif_label']
    
    print(f"\n{'='*80}")
    print("✅ Labeling Complete!")
    print(f"{'='*80}")
    print("Added columns to adata.obs:")
    if ploidy_config.get('enabled', True):
        print("  - ploidy_label, ploidy_label_str")
    print("  - prolif_label, prolif_label_str, prolif_marker_log")
    print("\n⚠️  Please review histograms before proceeding to prepare_data_for_elastic_net()")
    
    return adata

def apply_percentile_filter_from_subset(
    adata,
    feature: str,
    percentile: float,
    operator: str = 'lt',
    reference_subset: str = 'all',
    subset_label_key: Optional[str] = None,
    subset_label_value: Optional[int] = None,
    verbose: bool = True
):
    """
    Filter cells based on percentile threshold calculated from a reference subset.
    
    This implements the Fra1-filtering logic:
    1. Calculate threshold from a chosen subset (e.g., nonprolif cells)
    2. Apply threshold to ALL cells
    
    Parameters
    ----------
    adata : AnnData
        Data to filter
    feature : str
        Feature name to filter on (e.g., 'Int_Med_Fra1_nuc')
    percentile : float
        Percentile value (0-100)
    operator : str
        Comparison operator: 'lt', 'le', 'gt', 'ge', 'eq', 'ne'
        Default: 'lt' (less than, for "keep low Fra1 cells")
    reference_subset : str
        Which cells to use for calculating threshold:
        - 'all': All cells
        - 'nonprolif': Nonprolif cells only
        - 'prolif': Prolif cells only
        - '2N': 2N cells only
        Note: If filter_2n=True was applied first, these are implicitly 2N cells
    subset_label_key : str or None
        Column in adata.obs with labels (e.g., 'prolif_label')
        Required if reference_subset is 'nonprolif' or 'prolif'
    subset_label_value : int or None
        Value in subset_label_key for the reference subset
        - 0 for nonprolif
        - 1 for prolif
        
    Returns
    -------
    adata_filtered : AnnData
        Filtered cells
    threshold : float
        The threshold value used
    filter_info : dict
        Metadata about filtering:
        {
            'feature': str,
            'percentile': float,
            'threshold': float,
            'operator': str,
            'reference_subset': str,
            'n_reference_cells': int,
            'n_cells_before': int,
            'n_cells_after': int,
            'n_cells_removed': int,
            'fraction_removed': float
        }
        
    Examples
    --------
    >>> # Keep cells with Fra1 < P90 of nonprolif cells
    >>> adata_filt, thresh, info = apply_percentile_filter_from_subset(
    ...     adata,
    ...     feature='Int_Med_Fra1_nuc',
    ...     percentile=90,
    ...     operator='lt',
    ...     reference_subset='nonprolif',
    ...     subset_label_key='prolif_label',
    ...     subset_label_value=0
    ... )
    """
    if verbose:
        print(f"\n{'='*80}")
        print(f"Applying Percentile Filter from Subset")
        print(f"{'='*80}")
    
    # Select reference subset for threshold calculation
    if reference_subset == 'all':
        reference_cells = adata
    elif reference_subset == 'nonprolif':
        if subset_label_key is None or subset_label_value is None:
            raise ValueError("Must provide subset_label_key and subset_label_value for 'nonprolif' subset")
        reference_cells = adata[adata.obs[subset_label_key] == subset_label_value]
    elif reference_subset == 'prolif':
        if subset_label_key is None or subset_label_value is None:
            raise ValueError("Must provide subset_label_key and subset_label_value for 'prolif' subset")
        reference_cells = adata[adata.obs[subset_label_key] == 1]  # Assuming 1 = prolif
    elif reference_subset == '2N':
        reference_cells = adata[adata.obs['ploidy_label'] == 1]
    else:
        raise ValueError(f"Unknown reference_subset: {reference_subset}")
    
    # Calculate threshold from reference subset using obs_vector
    threshold = np.percentile(reference_cells.obs_vector(feature), percentile)
    
    if verbose:
        print(f"Feature: {feature}")
        print(f"Reference subset: {reference_subset} ({len(reference_cells):,} cells)")
        print(f"Percentile: P{percentile}")
        print(f"Threshold: {threshold:.4f}")
        print(f"Operator: {operator}")
    
    # Apply filter to ALL cells using obs_vector
    operators = {
        'lt': lambda x: x < threshold,
        'le': lambda x: x <= threshold,
        'gt': lambda x: x > threshold,
        'ge': lambda x: x >= threshold,
        'eq': lambda x: x == threshold,
        'ne': lambda x: x != threshold,
    }
    
    if operator not in operators:
        raise ValueError(f"Unknown operator: {operator}. Use: {list(operators.keys())}")
    
    mask = operators[operator](adata.obs_vector(feature))
    adata_filtered = adata[mask].copy()
    
    # Prepare filter info
    filter_info = {
        'feature': feature,
        'percentile': percentile,
        'threshold': threshold,
        'operator': operator,
        'reference_subset': reference_subset,
        'n_reference_cells': len(reference_cells),
        'n_cells_before': adata.n_obs,
        'n_cells_after': adata_filtered.n_obs,
        'n_cells_removed': adata.n_obs - adata_filtered.n_obs,
        'fraction_removed': (adata.n_obs - adata_filtered.n_obs) / adata.n_obs
    }
    
    if verbose:
        print(f"\nFilter results:")
        print(f"  Cells before: {filter_info['n_cells_before']:,}")
        print(f"  Cells after: {filter_info['n_cells_after']:,}")
        print(f"  Cells removed: {filter_info['n_cells_removed']:,} ({filter_info['fraction_removed']*100:.1f}%)")
        print(f"{'='*80}")
    
    return adata_filtered, threshold, filter_info

def prepare_data_for_elastic_net(
    adata,
    filter_2n: bool = True,
    optional_filter: Optional[Dict] = None,
    protein_list: Optional[List[str]] = None,
    exclude_conditions: Optional[List[str]] = None,
    verbose: bool = True,
    results_manager: Optional['ElasticNetResultsManager'] = None
):
    """
    Prepare data for Elastic Net analysis (Stage 2 preprocessing).
    
    This function assumes adata already has GMM labels from label_ploidy_and_proliferation_gmm().
    
    Pipeline:
    1. Exclude specified conditions (optional)
    2. Filter to 2N cells (optional, recommended)
    3. Apply percentile-based filter (optional, e.g., Fra1)
    4. Extract specified proteins
    
    Parameters
    ----------
    adata : AnnData
        Data with GMM labels already applied (from label_ploidy_and_proliferation_gmm)
        Must have columns: 'ploidy_label', 'prolif_label'
    filter_2n : bool
        If True, keep only 2N cells (default True)
    optional_filter : dict or None
        Percentile filter configuration:
        {
            'enabled': bool,
            'feature': str (e.g., 'Int_Med_Fra1_nuc'),
            'operator': str ('lt', 'le', 'gt', 'ge'),
            'percentile': float (0-100),
            'reference_subset': str ('all', 'nonprolif', 'prolif', '2N'),
            'subset_label_key': str (e.g., 'prolif_label'),
            'subset_label_value': int (e.g., 0 for nonprolif)
        }
    protein_list : list of str or None
        Proteins to extract (e.g., ['CycD1', 'CycE', 'p21', 'p27', 'CDK2', 'CDK4', 'CDK6'])
        If None, extracts all proteins in .X
    exclude_conditions : list of str or None
        Conditions to exclude (e.g., ['ERK2000'])
    verbose : bool
        Print progress information (default True)
    results_manager : ElasticNetResultsManager or None
        If provided, saves preprocessing info to manager.add_preprocessing_info()
        
    Returns
    -------
    adata_processed : AnnData
        Processed data ready for Elastic Net:
        - Filtered by ploidy, optional filter
        - Contains only specified proteins in .X (raw intensities)
        - Has 'prolif_label' column (binary: 0=nonprolif, 1=prolif)
    preprocessing_info : dict
        Metadata about preprocessing steps:
        {
            'n_cells_original': int,
            'n_cells_after_condition_filter': int,
            'n_cells_after_2n_filter': int,
            'n_cells_after_optional_filter': int,
            'n_cells_final': int,
            'filter_2n_applied': bool,
            'optional_filter_applied': bool,
            'optional_filter_info': dict or None,
            'excluded_conditions': list,
            'protein_list': list,
            'prolif_distribution': dict,  # {0: count, 1: count}
            'prolif_distribution_pct': dict  # {0: pct, 1: pct}
        }
        
    Examples
    --------
    >>> # With results manager (recommended for full workflow)
    >>> manager = ElasticNetResultsManager(cell_line='CFPAC', condition='control', additional_info='g0g1')
    >>> adata_prep, info = prepare_data_for_elastic_net(
    ...     adata,
    ...     filter_2n=True,
    ...     optional_filter={...},
    ...     protein_list=['CycD1', 'CycE', 'p21', 'p27', 'CDK2', 'CDK4', 'CDK6'],
    ...     results_manager=manager  # Auto-saves preprocessing info
    ... )
    >>> 
    >>> # Without manager (manual exploration)
    >>> adata_prep, info = prepare_data_for_elastic_net(
    ...     adata,
    ...     filter_2n=True,
    ...     protein_list=['CycD1', 'CycE', 'p21', 'p27', 'CDK2', 'CDK4', 'CDK6']
    ... )
    >>> 
    >>> # Step 3: Run Elastic Net
    >>> X_raw = adata_prep.X
    >>> y = adata_prep.obs['prolif_label'].values
    >>> results = evaluate_all_feature_sets(adata_prep, y, ...)
    """
    if verbose:
        print(f"\n{'='*80}")
        print(f"Preparing Data for Elastic Net Analysis")
        print(f"{'='*80}")
    
    # Check required columns
    if filter_2n and 'ploidy_label' not in adata.obs.columns:
        raise ValueError(
            "Cannot filter by 2N cells: 'ploidy_label' column not found!\n"
            "You must enable ploidy labeling in label_ploidy_and_proliferation_gmm().\n"
            "Options:\n"
            "  1. Re-run label_ploidy_and_proliferation_gmm() with ploidy_config={'enabled': True}\n"
            "  2. Set filter_2n=False in this function to skip ploidy filtering"
        )
    
    if 'prolif_label' not in adata.obs.columns:
        raise ValueError(
            "Missing 'prolif_label' column!\n"
            "Please run label_ploidy_and_proliferation_gmm() first!"
        )
    
    # Initialize tracking
    n_original = adata.n_obs
    adata_working = adata.copy()
    
    # ========================================================================
    # Step 1: Exclude conditions (optional)
    # ========================================================================
    if exclude_conditions is not None and len(exclude_conditions) > 0:
        # Check for both 'Condition' and 'condition' columns
        condition_col = None
        if 'Condition' in adata_working.obs.columns:
            condition_col = 'Condition'
        elif 'condition' in adata_working.obs.columns:
            condition_col = 'condition'
        
        if condition_col is not None:
            mask = ~adata_working.obs[condition_col].isin(exclude_conditions)
            adata_working = adata_working[mask].copy()
            if verbose:
                print(f"\nExcluded conditions: {exclude_conditions}")
                print(f"  Cells after exclusion: {adata_working.n_obs:,}")
        else:
            if verbose:
                print(f"\n⚠️  'Condition' or 'condition' column not found, skipping condition exclusion")
    
    n_after_condition = adata_working.n_obs
    
    # ========================================================================
    # Step 2: Filter to 2N cells (optional)
    # ========================================================================
    if filter_2n:
        if verbose:
            print(f"\nFiltering to 2N cells...")
        mask_2n = adata_working.obs['ploidy_label'] == 1
        adata_working = adata_working[mask_2n].copy()
        if verbose:
            print(f"  2N cells: {adata_working.n_obs:,}")
    else:
        if verbose:
            print(f"\n⚠️  2N filtering disabled - using all ploidy states")
    
    n_after_2n = adata_working.n_obs
    
    # ========================================================================
    # Step 3: Apply optional percentile filter (e.g., Fra1)
    # ========================================================================
    optional_filter_info = None
    if optional_filter is not None and optional_filter.get('enabled', False):
        adata_working, threshold, optional_filter_info = apply_percentile_filter_from_subset(
            adata_working,
            feature=optional_filter['feature'],
            percentile=optional_filter['percentile'],
            operator=optional_filter.get('operator', 'lt'),
            reference_subset=optional_filter.get('reference_subset', 'all'),
            subset_label_key=optional_filter.get('subset_label_key'),
            subset_label_value=optional_filter.get('subset_label_value'),
            verbose=verbose
        )
    else:
        if verbose:
            print(f"\n⚠️  Optional filter disabled - no percentile filtering applied")
    
    n_after_optional = adata_working.n_obs
    
    # ========================================================================
    # Step 4: Extract proteins (optional)
    # ========================================================================
    if protein_list is not None:
        if verbose:
            print(f"\nExtracting proteins: {protein_list}")
        
        # Check which proteins are available
        available_proteins = [p for p in protein_list if p in adata_working.var_names]
        missing_proteins = [p for p in protein_list if p not in adata_working.var_names]
        
        if missing_proteins:
            print(f"  ⚠️  Missing proteins (skipping): {missing_proteins}")
        
        if len(available_proteins) == 0:
            raise ValueError(f"No proteins from list found in data: {protein_list}")
        
        # Subset to available proteins
        adata_working = adata_working[:, available_proteins].copy()
        
        if verbose:
            print(f"  Extracted proteins: {available_proteins}")
            print(f"  Final protein count: {adata_working.n_vars}")
    
    # ========================================================================
    # Step 5: Prepare metadata summary
    # ========================================================================
    prolif_dist = adata_working.obs['prolif_label'].value_counts().to_dict()
    prolif_dist_pct = {k: v/adata_working.n_obs*100 for k, v in prolif_dist.items()}
    
    preprocessing_info = {
        'n_cells_original': n_original,
        'n_cells_after_condition_filter': n_after_condition,
        'n_cells_after_2n_filter': n_after_2n,
        'n_cells_after_optional_filter': n_after_optional,
        'n_cells_final': adata_working.n_obs,
        'filter_2n_applied': filter_2n,
        'optional_filter_applied': optional_filter is not None and optional_filter.get('enabled', False),
        'optional_filter_info': optional_filter_info,
        'excluded_conditions': exclude_conditions if exclude_conditions else [],
        'protein_list': protein_list if protein_list else list(adata_working.var_names),
        'prolif_distribution': prolif_dist,
        'prolif_distribution_pct': prolif_dist_pct
    }
    
    if verbose:
        print(f"\n{'='*80}")
        print(f"✅ Data Preparation Complete!")
        print(f"{'='*80}")
        print(f"Preprocessing summary:")
        print(f"  Original cells: {n_original:,}")
        print(f"  After condition filter: {n_after_condition:,}")
        print(f"  After 2N filter: {n_after_2n:,}")
        print(f"  After optional filter: {n_after_optional:,}")
        print(f"  Final cell count: {adata_working.n_obs:,}")
        print(f"\nProliferation distribution:")
        for label, count in prolif_dist.items():
            label_str = 'prolif' if label == 1 else 'nonprolif'
            print(f"  {label_str}: {count:,} ({prolif_dist_pct[label]:.1f}%)")
        print(f"\nReady for Elastic Net analysis!")
        print(f"  X = adata_prep.X  (raw protein intensities)")
        print(f"  y = adata_prep.obs['prolif_label'].values  (binary labels)")
    
    # Save preprocessing info to manager if provided
    if results_manager is not None:
        results_manager.add_preprocessing_info(preprocessing_info)
        if verbose:
            print(f"\n✅ Preprocessing info saved to results_manager")
    
    return adata_working, preprocessing_info
