"""
Visualization Module for Elastic Net Analysis Pipeline

This module provides professional, publication-ready diagnostic plots for
the Elastic Net cell cycle analysis pipeline.

Functions:
    - plot_model_selection(): Model comparison with 1-SE threshold
    - plot_coefficient_stability(): Coefficient consistency across folds
    - plot_feature_importance(): Final model coefficient bar chart
    - plot_threshold_sweep(): MCC optimization curve
    - plot_null_distribution(): Null model test visualization
    - plot_bootstrap_distribution(): Bootstrap CI visualization
    - plot_performance_summary(): AUROC vs features panel
    - create_summary_report(): Comprehensive HTML report (future)

Design Philosophy:
    - Input flexibility: Accept results dicts, DataFrames, or arrays
    - Consistent styling: Professional matplotlib/seaborn theme
    - Save capability: All functions support save_path parameter
    - Publication ready: High DPI, tight layouts, clear fonts

Author: AI Assistant
Date: November 13, 2025
"""

def plot_model_selection(
    results_df: pd.DataFrame,
    selection_results: Dict,
    save_path: Optional[Union[str, Path]] = None,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    figsize: Tuple[int, int] = (12, 6)
) -> Tuple[plt.Figure, plt.Axes]:
    """
    Create bar chart showing all models with 1-SE threshold line.
    
    Recreates the chart from the HTML protocol, showing all models
    sorted by AUROC with error bars and selection highlighting.
    
    Parameters
    ----------
    results_df : pd.DataFrame
        Output from evaluate_all_feature_sets()
        Must have: model_id, model_name, mean_auroc, se_auroc
    selection_results : dict
        Output from apply_1se_rule()
        Must have: best_model, primary_model, threshold
    save_path : str or None
        If provided, save figure to this path (overrides results_manager)
    results_manager : ElasticNetResultsManager or None
        If provided and save_path is None, auto-saves via manager.save_plot()
    figsize : tuple
        Figure size (width, height) in inches
        
    Returns
    -------
    fig, ax : matplotlib figure and axis
    
    Visual Elements
    ---------------
    - Bar chart sorted by AUROC (descending)
    - Error bars showing ±1 SE
    - Red star marking best model
    - Green star marking selected model (1-SE rule)
    - Horizontal dashed line at 1-SE threshold
    - Color-coded bars: gray (all), highlighted for best/selected
    
    Example
    -------
    >>> results_df = evaluate_all_feature_sets(...)
    >>> selection = apply_1se_rule(results_df)
    >>> fig, ax = plot_model_selection(results_df, selection, 
    ...                                  save_path='model_selection.png')
    >>> plt.show()
    """
    # Sort by AUROC for better visualization
    plot_df = results_df.sort_values('mean_auroc', ascending=False).reset_index(drop=True)
    
    # Extract info
    best_model = selection_results['best_model']
    selected_model = selection_results['primary_model']
    threshold = selection_results['threshold']
    
    # Create figure
    fig, ax = plt.subplots(figsize=figsize)
    
    # Plot all models with error bars
    x = np.arange(len(plot_df))
    ax.errorbar(
        x, plot_df['mean_auroc'], 
        yerr=plot_df['se_auroc'],
        fmt='o', capsize=5, markersize=6, 
        color='gray', alpha=0.6, 
        label='All models'
    )
    
    # Highlight best model
    best_idx = plot_df[plot_df['model_id'] == best_model['model_id']].index[0]
    ax.plot(
        best_idx, best_model['mean_auroc'], 
        'r*', markersize=20, 
        label=f"Best model (ID {best_model['model_id']})"
    )
    
    # Highlight selected model
    selected_idx = plot_df[plot_df['model_id'] == selected_model['model_id']].index[0]
    ax.plot(
        selected_idx, selected_model['mean_auroc'], 
        'g*', markersize=20, 
        label=f"Selected by 1-SE rule (ID {selected_model['model_id']})"
    )
    
    # Draw threshold line
    ax.axhline(
        threshold, color='red', linestyle='--', 
        alpha=0.7, linewidth=2,
        label=f'1-SE Threshold ({threshold:.3f})'
    )
    
    # Labels and formatting
    ax.set_xlabel('Models (sorted by AUROC, descending)', fontsize=11)
    ax.set_ylabel('Mean AUROC', fontsize=11)
    ax.set_title('Model Selection with 1-SE Rule', fontsize=12, fontweight='bold')
    ax.legend(loc='lower left', fontsize=9)
    ax.grid(True, alpha=0.3)
    
    # Add annotation with key info
    info_text = (
        f"Best: {best_model['model_name']}\n"
        f"  ({best_model['n_base_features']} features, "
        f"AUROC={best_model['mean_auroc']:.3f})\n\n"
        f"Selected: {selected_model['model_name']}\n"
        f"  ({selected_model['n_base_features']} features, "
        f"AUROC={selected_model['mean_auroc']:.3f})"
    )
    ax.text(
        0.98, 0.02, info_text,
        transform=ax.transAxes,
        fontsize=8,
        verticalalignment='bottom',
        horizontalalignment='right',
        bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.8)
    )
    
    plt.tight_layout()
    
    # Save logic: save_path takes priority, then results_manager
    if save_path is not None:
        plt.savefig(save_path, dpi=150, bbox_inches='tight')
        print(f"✓ Model selection plot saved to {save_path}")
    elif results_manager is not None:
        results_manager.save_plot(fig, 'model_selection')
        print(f"✓ Model selection plot saved via results_manager")
    
    return fig, ax

def plot_coefficient_stability(
    stability_df: pd.DataFrame,
    save_path: Optional[Union[str, Path]] = None,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    figsize: Tuple[int, int] = (12, 8)
) -> Tuple[plt.Figure, plt.Axes]:
    """
    Create horizontal bar chart with error bars showing coefficient stability.
    
    Color-codes features by stability (green=stable, red=unstable) and
    shows mean ± std dev across CV folds.
    
    Parameters
    ----------
    stability_df : pd.DataFrame
        Output from coefficient_stability_check()
        Must have: feature, mean_coef, std_coef, is_stable
    save_path : str or None
        If provided, save figure to this path (overrides results_manager)
    results_manager : ElasticNetResultsManager or None
        If provided and save_path is None, auto-saves via manager.save_plot()
    figsize : tuple
        Figure size (width, height)
        
    Returns
    -------
    fig, ax : matplotlib figure and axis
    
    Visual Elements
    ---------------
    - Horizontal bars sorted by |mean_coef| (most important first)
    - Error bars showing ±1 std dev
    - Green bars: stable features (all folds same sign)
    - Red bars: unstable features (sign flipping)
    - Vertical line at zero
    
    Example
    -------
    >>> stability_df = coefficient_stability_check(...)
    >>> fig, ax = plot_coefficient_stability(stability_df,
    ...                                        save_path='coef_stability.png')
    """
    # Sort by absolute mean coefficient
    stability_df_sorted = stability_df.sort_values('mean_coef', key=abs, ascending=True)
    
    # Create figure
    fig, ax = plt.subplots(figsize=figsize)
    
    # Create colors: green for stable, red for unstable
    colors = ['green' if stable else 'red' for stable in stability_df_sorted['is_stable']]
    
    # Plot mean coefficients with error bars (±1 std)
    y_pos = np.arange(len(stability_df_sorted))
    ax.barh(
        y_pos, 
        stability_df_sorted['mean_coef'],
        xerr=stability_df_sorted['std_coef'],
        color=colors, 
        alpha=0.7, 
        capsize=5
    )
    
    # Add labels
    ax.set_yticks(y_pos)
    ax.set_yticklabels(stability_df_sorted['feature'], fontsize=9)
    ax.axvline(0, color='black', linestyle='-', linewidth=0.8)
    ax.set_xlabel('Mean Coefficient ± Std Dev', fontsize=11)
    ax.set_title(
        'Coefficient Stability Across CV Folds\n(Green = Stable, Red = Unstable)',
        fontsize=12, fontweight='bold'
    )
    ax.grid(True, alpha=0.3, axis='x')
    
    # Add legend
    legend_elements = [
        Patch(facecolor='green', alpha=0.7, label='Stable (consistent sign)'),
        Patch(facecolor='red', alpha=0.7, label='Unstable (sign flipping)')
    ]
    ax.legend(handles=legend_elements, loc='best', fontsize=9)
    
    # Add summary statistics
    n_stable = stability_df['is_stable'].sum()
    n_unstable = (~stability_df['is_stable']).sum()
    summary_text = f"Stable: {n_stable}\nUnstable: {n_unstable}"
    ax.text(
        0.98, 0.02, summary_text,
        transform=ax.transAxes,
        fontsize=9,
        verticalalignment='bottom',
        horizontalalignment='right',
        bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.8)
    )
    
    plt.tight_layout()
    
    # Save logic: save_path takes priority, then results_manager
    if save_path is not None:
        plt.savefig(save_path, dpi=150, bbox_inches='tight')
        print(f"✓ Coefficient stability plot saved to {save_path}")
    elif results_manager is not None:
        results_manager.save_plot(fig, 'coefficient_stability')
        print(f"✓ Coefficient stability plot saved via results_manager")
    
    return fig, ax

def plot_feature_importance(
    final_classifier: Dict,
    save_path: Optional[Union[str, Path]] = None,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    figsize: Tuple[int, int] = (10, 6)
) -> Tuple[plt.Figure, plt.Axes]:
    """
    Bar chart of final model coefficients.
    
    Shows which features are most important for prediction.
    Positive coefficients increase proliferation probability.
    Negative coefficients decrease proliferation probability.
    
    Parameters
    ----------
    final_classifier : dict
        Output from build_final_classifier()
        Must have: 'feature_names', 'model' (with .coef_ attribute)
    save_path : str or None
        If provided, save figure to this path (overrides results_manager)
    results_manager : ElasticNetResultsManager or None
        If provided and save_path is None, auto-saves via manager.save_plot()
    figsize : tuple
        Figure size
        
    Returns
    -------
    fig, ax : matplotlib figure and axis
    
    Visual Elements
    ---------------
    - Horizontal bar chart sorted by |coefficient|
    - Green bars: positive coefficients (pro-proliferation)
    - Red bars: negative coefficients (anti-proliferation)
    - Vertical line at zero
    
    Example
    -------
    >>> final_classifier = build_final_classifier(...)
    >>> fig, ax = plot_feature_importance(
    ...     final_classifier,
    ...     save_path='feature_importance.png'
    ... )
    """
    # Extract coefficients and feature names
    coefficients = final_classifier['model'].coef_[0]
    feature_names = final_classifier['feature_names']
    
    # Create DataFrame and sort by coefficient value
    coef_df = pd.DataFrame({
        'feature': feature_names,
        'coefficient': coefficients
    })
    coef_df_sorted = coef_df.sort_values('coefficient', ascending=True)
    
    # Create figure
    fig, ax = plt.subplots(figsize=figsize)
    
    # Create colors: red for negative, green for positive
    colors = ['red' if x < 0 else 'green' for x in coef_df_sorted['coefficient']]
    
    # Plot horizontal bars
    ax.barh(
        range(len(coef_df_sorted)), 
        coef_df_sorted['coefficient'],
        color=colors, 
        alpha=0.7
    )
    
    # Add labels
    ax.set_yticks(range(len(coef_df_sorted)))
    ax.set_yticklabels(coef_df_sorted['feature'], fontsize=9)
    ax.axvline(0, color='black', linestyle='-', linewidth=0.8)
    ax.set_xlabel('Coefficient Value', fontsize=11)
    ax.set_title('Feature Importance (Model Coefficients)', fontsize=12, fontweight='bold')
    ax.grid(True, alpha=0.3, axis='x')
    
    # Add legend explaining colors
    legend_elements = [
        Patch(facecolor='green', alpha=0.7, label='Pro-proliferation (positive)'),
        Patch(facecolor='red', alpha=0.7, label='Anti-proliferation (negative)')
    ]
    ax.legend(handles=legend_elements, loc='best', fontsize=9)
    
    # Add interpretation note
    note_text = "Larger |coefficient| = stronger influence on prediction"
    ax.text(
        0.98, 0.02, note_text,
        transform=ax.transAxes,
        fontsize=8,
        verticalalignment='bottom',
        horizontalalignment='right',
        bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.8)
    )
    
    plt.tight_layout()
    
    # Save logic: save_path takes priority, then results_manager
    if save_path is not None:
        plt.savefig(save_path, dpi=150, bbox_inches='tight')
        print(f"✓ Feature importance plot saved to {save_path}")
    elif results_manager is not None:
        results_manager.save_plot(fig, 'feature_importance')
        print(f"✓ Feature importance plot saved via results_manager")
    
    return fig, ax

def plot_threshold_sweep(
    threshold_curve: Dict,
    optimal_threshold: float,
    save_path: Optional[Union[str, Path]] = None,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    figsize: Tuple[int, int] = (10, 6)
) -> Tuple[plt.Figure, plt.Axes]:
    """
    Plot metric value vs threshold, highlighting optimal threshold.
    
    Parameters
    ----------
    threshold_curve : dict
        From find_optimal_threshold() output or final_classifier['threshold_curve']
        Must have: 'thresholds', 'mcc_values' (or other metric)
    optimal_threshold : float
        The selected optimal threshold value
    save_path : str or None
        If provided, save figure to this path (overrides results_manager)
    results_manager : ElasticNetResultsManager or None
        If provided and save_path is None, auto-saves via manager.save_plot()
    figsize : tuple
        Figure size
        
    Returns
    -------
    fig, ax : matplotlib figure and axis
    
    Visual Elements
    ---------------
    - Line plot: threshold vs metric (e.g., MCC)
    - Vertical red dashed line at optimal threshold
    - Horizontal green dashed line at max metric value
    - Annotation box with optimal values
    
    Example
    -------
    >>> from build_final_classifier output:
    >>> fig, ax = plot_threshold_sweep(
    ...     final_classifier['threshold_curve'],
    ...     final_classifier['optimal_threshold'],
    ...     save_path='threshold_sweep.png'
    ... )
    """
    # Extract data
    thresholds = threshold_curve['thresholds']
    mcc_values = threshold_curve['mcc_values']
    max_mcc = np.max(mcc_values)
    
    # Create figure
    fig, ax = plt.subplots(figsize=figsize)
    
    # Plot threshold curve
    ax.plot(thresholds, mcc_values, linewidth=2, color='blue', label='MCC vs Threshold')
    
    # Mark optimal threshold
    ax.axvline(
        optimal_threshold, 
        color='red', 
        linestyle='--',
        linewidth=2,
        label=f"Optimal threshold: {optimal_threshold:.3f}"
    )
    
    # Mark max MCC
    ax.axhline(
        max_mcc, 
        color='green', 
        linestyle='--',
        alpha=0.5, 
        linewidth=2,
        label=f"Max MCC: {max_mcc:.3f}"
    )
    
    # Labels and formatting
    ax.set_xlabel('Classification Threshold', fontsize=11)
    ax.set_ylabel('Matthews Correlation Coefficient (MCC)', fontsize=11)
    ax.set_title('Threshold Optimization Curve', fontsize=12, fontweight='bold')
    ax.legend(loc='best', fontsize=9)
    ax.grid(True, alpha=0.3)
    
    # Add annotation box
    info_text = (
        f"Optimal: {optimal_threshold:.3f}\n"
        f"MCC: {max_mcc:.3f}"
    )
    ax.text(
        0.02, 0.98, info_text,
        transform=ax.transAxes,
        fontsize=10,
        verticalalignment='top',
        horizontalalignment='left',
        bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.8)
    )
    
    plt.tight_layout()
    
    # Save logic: save_path takes priority, then results_manager
    if save_path is not None:
        plt.savefig(save_path, dpi=150, bbox_inches='tight')
        print(f"✓ Threshold sweep plot saved to {save_path}")
    elif results_manager is not None:
        results_manager.save_plot(fig, 'threshold_sweep')
        print(f"✓ Threshold sweep plot saved via results_manager")
    
    return fig, ax

def plot_null_distribution(
    null_results: Dict,
    save_path: Optional[Union[str, Path]] = None,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    figsize: Tuple[int, int] = (10, 6)
) -> Tuple[plt.Figure, plt.Axes]:
    """
    Plot null model test distribution.
    
    Shows histogram of AUROC values from shuffled labels (null distribution)
    compared to true model AUROC.
    
    Parameters
    ----------
    null_results : dict
        Output from null_model_test()
        Must have: true_model_auroc, null_aurocs, mean_null_auroc, p_value, is_significant
    save_path : str or None
        If provided, save figure to this path (overrides results_manager)
    results_manager : ElasticNetResultsManager or None
        If provided and save_path is None, auto-saves via manager.save_plot()
    figsize : tuple
        Figure size
        
    Returns
    -------
    fig, ax : matplotlib figure and axis
    
    Visual Elements
    ---------------
    - Histogram of null AUROCs (gray)
    - Vertical line at true AUROC (red)
    - Vertical line at mean null AUROC (blue)
    - Text box with p-value and significance
    
    Example
    -------
    >>> null_results = null_model_test(...)
    >>> fig, ax = plot_null_distribution(null_results, save_path='null_test.png')
    """
    fig, ax = plt.subplots(figsize=figsize)
    
    # Histogram of null AUROCs
    ax.hist(
        null_results['null_aurocs'], 
        bins=15, 
        alpha=0.7, 
        color='gray',
        edgecolor='black', 
        label='Null distribution (shuffled labels)'
    )
    
    # Mark true AUROC
    ax.axvline(
        null_results['true_model_auroc'], 
        color='red', 
        linestyle='--',
        linewidth=2, 
        label=f"True AUROC: {null_results['true_model_auroc']:.3f}"
    )
    
    # Mark mean null AUROC
    ax.axvline(
        null_results['mean_null_auroc'], 
        color='blue', 
        linestyle=':',
        linewidth=2, 
        label=f"Mean Null: {null_results['mean_null_auroc']:.3f}"
    )
    
    # Add text box with p-value
    sig_symbol = "✅" if null_results['is_significant'] else "❌"
    textstr = (
        f"p-value: {null_results['p_value']:.4f}\n"
        f"Significant: {null_results['is_significant']} {sig_symbol}\n"
        f"{'Model beats random' if null_results['is_significant'] else 'Model = random'}"
    )
    props = dict(boxstyle='round', facecolor='wheat', alpha=0.8)
    ax.text(
        0.05, 0.95, textstr, 
        transform=ax.transAxes, 
        fontsize=10,
        verticalalignment='top', 
        bbox=props
    )
    
    # Labels and formatting
    ax.set_xlabel('AUROC', fontsize=11)
    ax.set_ylabel('Frequency', fontsize=11)
    ax.set_title('Null Model Test: True Model vs. Shuffled Labels', 
                 fontsize=12, fontweight='bold')
    ax.legend(loc='upper right', fontsize=9)
    ax.grid(True, alpha=0.3)
    
    plt.tight_layout()
    
    # Save logic: save_path takes priority, then results_manager
    if save_path is not None:
        plt.savefig(save_path, dpi=150, bbox_inches='tight')
        print(f"✓ Null distribution plot saved to {save_path}")
    elif results_manager is not None:
        results_manager.save_plot(fig, 'null_distribution')
        print(f"✓ Null distribution plot saved via results_manager")
    
    return fig, ax

def plot_bootstrap_distribution(
    bootstrap_results: Dict,
    save_path: Optional[Union[str, Path]] = None,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    figsize: Tuple[int, int] = (10, 6)
) -> Tuple[plt.Figure, plt.Axes]:
    """
    Plot bootstrap confidence interval distribution.
    
    Shows histogram of bootstrap AUROC values with confidence intervals
    and summary statistics.
    
    Parameters
    ----------
    bootstrap_results : dict
        Output from bootstrap_confidence_intervals()
        Must have: bootstrap_aurocs, mean_auroc, median_auroc, ci_lower, 
                   ci_upper, n_bootstrap
    save_path : str or None
        If provided, save figure to this path (overrides results_manager)
    results_manager : ElasticNetResultsManager or None
        If provided and save_path is None, auto-saves via manager.save_plot()
    figsize : tuple
        Figure size
        
    Returns
    -------
    fig, ax : matplotlib figure and axis
    
    Visual Elements
    ---------------
    - Histogram of bootstrap AUROCs (skyblue)
    - Vertical line at mean (red)
    - Vertical line at median (orange)
    - Vertical lines at CI bounds (green)
    - Text box with CI width and interpretation
    
    Example
    -------
    >>> bootstrap_results = bootstrap_confidence_intervals(...)
    >>> fig, ax = plot_bootstrap_distribution(bootstrap_results, 
    ...                                         save_path='bootstrap_ci.png')
    """
    fig, ax = plt.subplots(figsize=figsize)
    
    # Histogram of bootstrap AUROCs
    ax.hist(
        bootstrap_results['bootstrap_aurocs'], 
        bins=20, 
        alpha=0.7,
        color='skyblue',
        edgecolor='black', 
        label='Bootstrap distribution'
    )
    
    # Mark mean and median
    ax.axvline(
        bootstrap_results['mean_auroc'], 
        color='red', 
        linestyle='--',
        linewidth=2, 
        label=f"Mean: {bootstrap_results['mean_auroc']:.3f}"
    )
    ax.axvline(
        bootstrap_results['median_auroc'], 
        color='orange', 
        linestyle=':',
        linewidth=2, 
        label=f"Median: {bootstrap_results['median_auroc']:.3f}"
    )
    
    # Mark confidence interval
    ax.axvline(
        bootstrap_results['ci_lower'], 
        color='green', 
        linestyle='--',
        linewidth=2, 
        label=f"95% CI: [{bootstrap_results['ci_lower']:.3f}, {bootstrap_results['ci_upper']:.3f}]"
    )
    ax.axvline(
        bootstrap_results['ci_upper'], 
        color='green', 
        linestyle='--',
        linewidth=2
    )
    
    # Add text box with CI width
    ci_width = bootstrap_results['ci_upper'] - bootstrap_results['ci_lower']
    stability_text = 'Narrow (stable)' if ci_width < 0.1 else 'Wide (unstable)'
    stability_symbol = '✅' if ci_width < 0.1 else '⚠️'
    textstr = (
        f"CI Width: {ci_width:.3f}\n"
        f"{stability_text} {stability_symbol}"
    )
    props = dict(boxstyle='round', facecolor='wheat', alpha=0.8)
    ax.text(
        0.05, 0.95, textstr, 
        transform=ax.transAxes, 
        fontsize=10,
        verticalalignment='top', 
        bbox=props
    )
    
    # Labels and formatting
    ax.set_xlabel('AUROC', fontsize=11)
    ax.set_ylabel('Frequency', fontsize=11)
    ax.set_title(
        f'Bootstrap Distribution ({bootstrap_results["n_bootstrap"]} samples)',
        fontsize=12, fontweight='bold'
    )
    ax.legend(loc='upper left', fontsize=9)
    ax.grid(True, alpha=0.3)
    
    plt.tight_layout()
    
    # Save logic: save_path takes priority, then results_manager
    if save_path is not None:
        plt.savefig(save_path, dpi=150, bbox_inches='tight')
        print(f"✓ Bootstrap distribution plot saved to {save_path}")
    elif results_manager is not None:
        results_manager.save_plot(fig, 'bootstrap_distribution')
        print(f"✓ Bootstrap distribution plot saved via results_manager")
    
    return fig, ax

def plot_performance_summary(
    results_df: pd.DataFrame,
    save_path: Optional[Union[str, Path]] = None,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    figsize: Tuple[int, int] = (14, 5)
) -> Tuple[plt.Figure, np.ndarray]:
    """
    Create summary panel with AUROC vs features and AUROC distribution.
    
    Two-panel figure showing:
    1. AUROC vs number of base features (scatter with error bars)
    2. AUROC distribution histogram
    
    Parameters
    ----------
    results_df : pd.DataFrame
        Output from evaluate_all_feature_sets()
        Must have: n_base_features, mean_auroc, se_auroc
    save_path : str or None
        If provided, save figure to this path (overrides results_manager)
    results_manager : ElasticNetResultsManager or None
        If provided and save_path is None, auto-saves via manager.save_plot()
    figsize : tuple
        Figure size (width, height)
        
    Returns
    -------
    fig, axes : matplotlib figure and array of axes
    
    Example
    -------
    >>> results_df = evaluate_all_feature_sets(...)
    >>> fig, axes = plot_performance_summary(results_df, 
    ...                                        save_path='performance_summary.png')
    """
    fig, axes = plt.subplots(1, 2, figsize=figsize)
    
    # Plot 1: AUROC vs Number of Features
    ax = axes[0]
    ax.errorbar(
        results_df['n_base_features'], 
        results_df['mean_auroc'],
        yerr=results_df['se_auroc'], 
        fmt='o', 
        capsize=5, 
        markersize=8,
        alpha=0.6
    )
    ax.set_xlabel('Number of Base Features', fontsize=11)
    ax.set_ylabel('Mean AUROC', fontsize=11)
    ax.set_title('Model Performance vs Feature Count', fontsize=12, fontweight='bold')
    ax.grid(True, alpha=0.3)
    
    # Plot 2: AUROC distribution
    ax = axes[1]
    results_df['mean_auroc'].hist(bins=10, ax=ax, edgecolor='black', alpha=0.7)
    ax.axvline(
        results_df['mean_auroc'].mean(), 
        color='red', 
        linestyle='--',
        linewidth=2,
        label=f"Mean: {results_df['mean_auroc'].mean():.3f}"
    )
    ax.set_xlabel('Mean AUROC', fontsize=11)
    ax.set_ylabel('Frequency', fontsize=11)
    ax.set_title('AUROC Distribution', fontsize=12, fontweight='bold')
    ax.legend(fontsize=9)
    ax.grid(True, alpha=0.3)
    
    plt.tight_layout()
    
    # Save logic: save_path takes priority, then results_manager
    if save_path is not None:
        plt.savefig(save_path, dpi=150, bbox_inches='tight')
        print(f"✓ Performance summary plot saved to {save_path}")
    elif results_manager is not None:
        results_manager.save_plot(fig, 'performance_summary')
        print(f"✓ Performance summary plot saved via results_manager")
    
    return fig, axes

def plot_scatter_kde(
    adata,
    x_feature: Union[str, Tuple[str, str]],
    y_feature: Union[str, Tuple[str, str]],
    label_key: str,
    x_threshold: Optional[float] = None,
    y_threshold: Optional[float] = None,
    save_path: Optional[Union[str, Path]] = None,
    results_manager: Optional['ElasticNetResultsManager'] = None,
    figsize: Tuple[int, int] = (10, 8),
    scatter_alpha: float = 0.5,
    kde_levels: int = 10,
    title: Optional[str] = None,
    xlim_percentile: Optional[Union[float, Tuple[float, float]]] = None,
    ylim_percentile: Optional[Union[float, Tuple[float, float]]] = None
) -> Tuple:
    """
    Create scatter plot with 2D KDE contours and marginal distributions.
    
    Visualizes relationship between two features (single or ratio) with
    proliferation labels, showing 2D KDE contours and optional thresholds.
    
    Parameters
    ----------
    adata : AnnData
        AnnData object containing features and labels
    x_feature : str or tuple
        X-axis feature. Single feature name or (numerator, denominator) for ratio
        Example: 'Int_Med_Fra1_nuc' or ('Int_Med_pRB_nuc', 'Int_Med_RB_nuc')
    y_feature : str or tuple
        Y-axis feature. Single feature name or (numerator, denominator) for ratio
    label_key : str
        Column name in adata.obs for binary labels (e.g., 'prolif_gmm_label')
    x_threshold : float or None
        Optional vertical line threshold for x-axis (e.g., filter cutoff)
    y_threshold : float or None
        Optional horizontal line threshold for y-axis
    save_path : str or None
        If provided, save figure to this path (overrides results_manager)
    results_manager : ElasticNetResultsManager or None
        If provided and save_path is None, auto-saves via manager.save_plot()
    figsize : tuple
        Figure size (width, height)
    scatter_alpha : float
        Transparency of scatter points (0-1)
    kde_levels : int
        Number of contour levels for 2D KDE
    title : str or None
        Custom title. If None, auto-generates from features
    xlim_percentile : float, tuple of floats, or None
        If float: set upper x-axis limit to this percentile, auto-scale lower (e.g., 99)
        If tuple: set (lower, upper) x-axis limits to these percentiles (e.g., (1, 99))
                 Can use None in tuple to skip a bound (e.g., (None, 99) for upper only)
        If None: auto-scale both limits (default)
    ylim_percentile : float, tuple of floats, or None
        If float: set upper y-axis limit to this percentile, auto-scale lower (e.g., 99)
        If tuple: set (lower, upper) y-axis limits to these percentiles (e.g., (1, 99))
                 Can use None in tuple to skip a bound (e.g., (None, 99) for upper only)
        If None: auto-scale both limits (default)
        
    Returns
    -------
    fig, axes : matplotlib figure and dict of axes
        axes dict has keys: 'main', 'top', 'right'
        
    Visual Elements
    ---------------
    Layout:
        - Main: Scatter plot with 2D KDE contours (colored by label)
        - Top: X-axis marginal distribution (1D KDE per label)
        - Right: Y-axis marginal distribution (1D KDE per label)
    
    Features:
        - Scatter points colored by label class
        - 2D KDE contours for each label class
        - Dashed red lines for thresholds (if provided)
        - Marginal 1D KDE distributions
        
    Example
    -------
    >>> # Single features with upper percentile limit only
    >>> fig, axes = plot_scatter_kde(
    ...     adata,
    ...     x_feature='Int_Med_Fra1_nuc',
    ...     y_feature='Int_Med_PCNA_nuc',
    ...     label_key='prolif_gmm_label',
    ...     x_threshold=2.5,
    ...     xlim_percentile=99  # Only set upper limit to 99th percentile
    ... )
    
    >>> # With both lower and upper percentile limits
    >>> fig, axes = plot_scatter_kde(
    ...     adata,
    ...     x_feature='Int_Med_Fra1_nuc',
    ...     y_feature='Int_Med_PCNA_nuc',
    ...     label_key='prolif_gmm_label',
    ...     xlim_percentile=(1, 99),  # Set both limits
    ...     ylim_percentile=(1, 99)
    ... )
    
    >>> # With None in tuple to specify only one bound
    >>> fig, axes = plot_scatter_kde(
    ...     adata,
    ...     x_feature='Int_Med_Fra1_nuc',
    ...     y_feature='Int_Med_PCNA_nuc',
    ...     label_key='prolif_gmm_label',
    ...     xlim_percentile=(None, 99),  # Only upper, same as xlim_percentile=99
    ...     ylim_percentile=(5, None)     # Only lower limit
    ... )
    
    >>> # Ratio feature with auto-save
    >>> fig, axes = plot_scatter_kde(
    ...     adata,
    ...     x_feature='Int_Med_Fra1_nuc',
    ...     y_feature=('Int_Med_pRB_nuc', 'Int_Med_RB_nuc'),
    ...     label_key='prolif_gmm_label',
    ...     results_manager=manager
    ... )
    """
    # Helper function to extract feature values
    def get_log10_feature_values(feature):
        """Extract feature values using obs_vector()."""
        if isinstance(feature, tuple):
            # Ratio: numerator / denominator
            num_values = adata.obs_vector(feature[0])
            denom_values = adata.obs_vector(feature[1])
            # Avoid division by zero
            values = np.divide(num_values, denom_values, 
                             where=denom_values != 0, 
                             out=np.full_like(num_values, np.nan))
            # Generate label
            label = f"{feature[0]} / {feature[1]}"
        else:
            # Single feature
            values = adata.obs_vector(feature)
            label = feature
        return values, label
    
    # Extract x and y values
    x_values, x_label = get_log10_feature_values(x_feature)
    y_values, y_label = get_log10_feature_values(y_feature)
    
    # Extract labels
    labels = adata.obs[label_key].values
    unique_labels = np.unique(labels)
    
    # Remove NaN values
    valid_mask = ~(np.isnan(x_values) | np.isnan(y_values))
    x_values = x_values[valid_mask]
    y_values = y_values[valid_mask]
    labels = labels[valid_mask]
    
    # Calculate axis limits based on percentiles if provided
    x_lim_lower, x_lim_upper = None, None
    if xlim_percentile is not None:
        if isinstance(xlim_percentile, tuple):
            # Tuple: (lower, upper) percentiles
            # Allow None in tuple to skip that bound
            x_lim_lower = np.percentile(x_values, xlim_percentile[0]) if xlim_percentile[0] is not None else None
            x_lim_upper = np.percentile(x_values, xlim_percentile[1]) if xlim_percentile[1] is not None else None
        else:
            # Single value: only upper limit, auto-scale lower
            x_lim_upper = np.percentile(x_values, xlim_percentile)
    
    y_lim_lower, y_lim_upper = None, None
    if ylim_percentile is not None:
        if isinstance(ylim_percentile, tuple):
            # Tuple: (lower, upper) percentiles
            # Allow None in tuple to skip that bound
            y_lim_lower = np.percentile(y_values, ylim_percentile[0]) if ylim_percentile[0] is not None else None
            y_lim_upper = np.percentile(y_values, ylim_percentile[1]) if ylim_percentile[1] is not None else None
        else:
            # Single value: only upper limit, auto-scale lower
            y_lim_upper = np.percentile(y_values, ylim_percentile)
    
    # Create figure with gridspec for marginals
    from matplotlib.gridspec import GridSpec
    fig = plt.figure(figsize=figsize)
    gs = GridSpec(4, 4, figure=fig, hspace=0.05, wspace=0.05)
    
    ax_main = fig.add_subplot(gs[1:, :-1])
    ax_top = fig.add_subplot(gs[0, :-1], sharex=ax_main)
    ax_right = fig.add_subplot(gs[1:, -1], sharey=ax_main)
    
    # Color map for labels
    colors = ['blue', 'orange']
    label_colors = {label: colors[i] for i, label in enumerate(unique_labels)}
    
    # Main scatter plot with KDE contours
    for label in unique_labels:
        mask = labels == label
        x_subset = x_values[mask]
        y_subset = y_values[mask]
        
        # Scatter plot
        ax_main.scatter(
            x_subset, y_subset,
            c=label_colors[label],
            alpha=scatter_alpha,
            s=20,
            label=str(label),
            edgecolors='none'
        )
        
        # 2D KDE contours (if enough points)
        if len(x_subset) > 10:
            try:
                # Calculate 2D KDE
                xy = np.vstack([x_subset, y_subset])
                kernel = stats.gaussian_kde(xy)
                
                # Create grid for evaluation
                x_min, x_max = x_values.min(), x_values.max()
                y_min, y_max = y_values.min(), y_values.max()
                xx, yy = np.mgrid[x_min:x_max:100j, y_min:y_max:100j]
                positions = np.vstack([xx.ravel(), yy.ravel()])
                zz = np.reshape(kernel(positions).T, xx.shape)
                
                # Plot contours
                ax_main.contour(
                    xx, yy, zz,
                    levels=kde_levels,
                    colors=label_colors[label],
                    alpha=0.6,
                    linewidths=1.5
                )
            except Exception as e:
                print(f"  Warning: Could not compute KDE for {label}: {e}")
    
    # Add threshold lines if provided
    if x_threshold is not None:
        ax_main.axvline(
            x_threshold,
            color='red',
            linestyle='--',
            linewidth=2,
            alpha=0.7,
            label=f'X threshold: {x_threshold:.3f}'
        )
        ax_top.axvline(x_threshold, color='red', linestyle='--', linewidth=2, alpha=0.7)
    
    if y_threshold is not None:
        ax_main.axhline(
            y_threshold,
            color='red',
            linestyle='--',
            linewidth=2,
            alpha=0.7,
            label=f'Y threshold: {y_threshold:.3f}'
        )
        ax_right.axhline(y_threshold, color='red', linestyle='--', linewidth=2, alpha=0.7)
    
    # Main plot formatting
    ax_main.set_xlabel(x_label, fontsize=10)
    ax_main.set_ylabel(y_label, fontsize=10)
    ax_main.legend(loc='best', fontsize=9)
    ax_main.grid(True, alpha=0.3)
    
    # Apply axis limits if percentiles were specified
    if x_lim_lower is not None or x_lim_upper is not None:
        ax_main.set_xlim(left=x_lim_lower, right=x_lim_upper)
    if y_lim_lower is not None or y_lim_upper is not None:
        ax_main.set_ylim(bottom=y_lim_lower, top=y_lim_upper)
    
    # Top marginal (X distribution)
    for label in unique_labels:
        mask = labels == label
        x_subset = x_values[mask]
        
        if len(x_subset) > 1:
            # 1D KDE
            kde = stats.gaussian_kde(x_subset)
            x_range = np.linspace(x_values.min(), x_values.max(), 200)
            ax_top.plot(
                x_range,
                kde(x_range),
                color=label_colors[label],
                linewidth=2,
                label=str(label)
            )
            ax_top.fill_between(
                x_range,
                kde(x_range),
                alpha=0.3,
                color=label_colors[label]
            )
    
    ax_top.set_ylabel('Density', fontsize=9)
    ax_top.tick_params(labelbottom=False)
    ax_top.grid(True, alpha=0.3)
    
    # Right marginal (Y distribution)
    for label in unique_labels:
        mask = labels == label
        y_subset = y_values[mask]
        
        if len(y_subset) > 1:
            # 1D KDE
            kde = stats.gaussian_kde(y_subset)
            y_range = np.linspace(y_values.min(), y_values.max(), 200)
            ax_right.plot(
                kde(y_range),
                y_range,
                color=label_colors[label],
                linewidth=2,
                label=str(label)
            )
            ax_right.fill_betweenx(
                y_range,
                kde(y_range),
                alpha=0.3,
                color=label_colors[label]
            )
    
    ax_right.set_xlabel('Density', fontsize=9)
    ax_right.tick_params(labelleft=False)
    ax_right.grid(True, alpha=0.3)
    
    # Title
    if title is None:
        title = f'Feature Scatter with KDE: {x_label} vs {y_label}'
    fig.suptitle(title, fontsize=12, fontweight='bold', y=0.98)
    
    plt.tight_layout()
    
    # Save logic: save_path takes priority, then results_manager
    if save_path is not None:
        plt.savefig(save_path, dpi=150, bbox_inches='tight')
        print(f"✓ Scatter KDE plot saved to {save_path}")
    elif results_manager is not None:
        results_manager.save_plot(fig, 'filter_scatter_kde')
        print(f"✓ Scatter KDE plot saved via results_manager")
    
    return fig, {'main': ax_main, 'top': ax_top, 'right': ax_right}
