"""
Model Selection Module for Elastic Net Analysis

This module implements model selection functions:
1. evaluate_all_feature_sets() - Run nested CV on all 128 feature sets
2. apply_1se_rule() - Select simplest model within 1 SE of best

4. find_optimal_threshold() - Maximize MCC on validation set

def evaluate_all_feature_sets(
    adata,
    y: np.ndarray,
    feature_combinations: List[Dict],
    param_grid: Dict,
    results_manager: 'ElasticNetResultsManager',
    outer_cv_splits: int = 10,
    outer_cv_repeats: int = 1,
    inner_cv: int = 5,
    random_state: int = 42,
    n_jobs: int = -1,
    verbose: int = 1,
    resume: bool = True,
    keep_checkpoints: bool = False,
    use_composite_features: bool = True,
    feature_mode: Optional[str] = None
) -> pd.DataFrame:
    """
    Evaluate all 128 feature set models using nested CV with parallelization.
    
    Parameters
    ----------
    adata : AnnData
        Single-cell data object with RAW protein intensities
    y : np.ndarray
        Binary labels (0/1)
    feature_combinations : list of dict
        Output from generate_all_feature_combinations()
        Each dict must have 'id' key (0-127) for consistent model identification
    param_grid : dict
        Hyperparameter grid for Elastic Net
        Example: {'C': [0.1, 1.0, 10.0], 'l1_ratio': [0.1, 0.5, 0.9]}
    results_manager : ElasticNetResultsManager
        Results manager instance for organizing all outputs.
        Automatically handles checkpoints, warnings, and final results saving.
        **REQUIRED** - no longer optional for clean, organized results.
    outer_cv_splits : int
        Number of splits for RepeatedStratifiedKFold outer loop (default 10)
    outer_cv_repeats : int
        Number of repeats for RepeatedStratifiedKFold outer loop (default 1)
    inner_cv : int
        Number of inner CV folds for GridSearchCV (default 5)
    random_state : int
        Random seed for reproducibility
    n_jobs : int
        Number of parallel jobs for feature set evaluation (-1 = all cores)
    verbose : int
        Verbosity level:
        - 0: Silent (no output)
        - 1: Feature set progress bar only (default)
        - 2: Feature set + nested fold progress bars
    resume : bool
        If True, skip feature sets that already have checkpoint files (default True)
    keep_checkpoints : bool
        If True, keep individual JSON checkpoint files after combining (default False)
        If False, remove checkpoints/ directory entirely after successful combination
    use_composite_features : bool
        Legacy parameter. Ignored if feature_mode is provided.
    feature_mode : str, optional
        'base', 'mixed', or 'composite'. If None, inferred from use_composite_features.
        
    Returns
    -------
    results_df : pd.DataFrame
        Also automatically saved via results_manager.save_model_results()
        Columns:
        - model_id: int (0-127, from feature_set_config['id'])
        - model_name: str (e.g., "CycD1_CDK4_p27")
        - features: list of str (proteins in this set)
        - n_base_features: int (number of proteins, 0-7)
        - n_total_features: int (proteins + 5 composites, varies after NaN removal)
        - mean_auroc: float (mean across outer folds)
        - se_auroc: float (standard error = SD / sqrt(n_folds))
        - sd_auroc: float (standard deviation of fold scores)
        - fold_scores: list of float (AUROC for each outer fold)
        - best_params: dict (most common hyperparameters across folds)
        - timestamp: str (ISO format)
        
    Examples
    --------
    >>> from elastic_net_modules import ElasticNetResultsManager
    >>> manager = ElasticNetResultsManager(
    ...     cell_line="CFPAC",
    ...     condition="control",
    ...     additional_info="g0g1"
    ... )
    >>> results_df = evaluate_all_feature_sets(
    ...     adata, y, feature_combos, param_grid,
    ...     results_manager=manager,
    ...     n_jobs=-1
    ... )
    """
    # Step 1: Initialize evaluation configuration
    # PURPOSE: Set up feature mode and directory paths for model evaluation
    # INPUTS: Feature mode preference, Results manager instance
    # OUTPUTS: Configured feature_mode, checkpoint and warnings directories
    
    # Backwards compatibility logic
    if feature_mode is None:
        feature_mode = 'mixed' if use_composite_features else 'base'
        
    # Get directory paths from manager
    checkpoints_dir = Path(results_manager.get_checkpoint_dir())
    warnings_dir = Path(results_manager.dirs['warnings'])
    
    if verbose > 0:
        print(f"Analysis ID: {results_manager.analysis_id}")
        print(f"Results directory: {results_manager.base_dir}")
        print(f"Checkpoints: {checkpoints_dir}")
        print(f"Warnings: {warnings_dir}")
        print(f"Feature Mode: {feature_mode}")
    
    # Step 2: Determine which feature sets to process
    # PURPOSE: Identify models to evaluate based on existing checkpoints and resume setting
    # INPUTS: Feature combinations (128 models), Checkpoint directory, Resume flag
    # OUTPUTS: List of feature sets requiring evaluation
    
    feature_sets_to_process = []
    for config in feature_combinations:
        model_id = config['id']
        checkpoint_path = checkpoints_dir / f"model_{model_id:03d}.json"
        
        if resume and checkpoint_path.exists():
            if verbose > 0:
                print(f"Skipping model {model_id} (checkpoint exists)")
            continue
        
        feature_sets_to_process.append(config)
    
    # Calculate counts for progress reporting
    n_total = len(feature_combinations)
    n_to_process = len(feature_sets_to_process)
    n_completed = n_total - n_to_process
    
    if verbose > 0:
        print(f"\nTotal models: {n_total}")
        print(f"Already completed: {n_completed}")
        print(f"To process: {n_to_process}")
    
    # Step 3: Evaluate feature sets in parallel
    # PURPOSE: Run nested cross-validation for each model using parallel processing
    # INPUTS: Feature sets to process, Data (adata, y), Hyperparameter grid, CV settings
    # OUTPUTS: Checkpoint files saved for each evaluated model
    # CRITICAL: This is the most time-intensive step - uses joblib parallelization
    # FUNCTION: _evaluate_single_feature_set
    
    # Process feature sets in parallel
    if n_to_process > 0:
        # Nested verbosity for individual nested_cv calls
        nested_verbose = verbose >= 2
        
        if verbose == 1:
            # Show progress bar for feature sets
            def process_with_progress(config):
                return _evaluate_single_feature_set(
                    adata, y, config, param_grid,
                    checkpoints_dir, outer_cv_splits, outer_cv_repeats,
                    inner_cv, random_state, nested_verbose, 
                    feature_mode=feature_mode
                )
            
            results = Parallel(n_jobs=n_jobs)(
                delayed(process_with_progress)(config)
                for config in tqdm(feature_sets_to_process, desc="Evaluating feature sets")
            )
        else:
            # Silent or verbose=2 (no outer progress bar)
            results = Parallel(n_jobs=n_jobs)(
                delayed(_evaluate_single_feature_set)(
                    adata, y, config, param_grid,
                    checkpoints_dir, outer_cv_splits, outer_cv_repeats,
                    inner_cv, random_state, nested_verbose, 
                    feature_mode=feature_mode
                )
                for config in feature_sets_to_process
            )
    
    # Step 4: Load and combine checkpoint results
    # PURPOSE: Aggregate all individual model evaluations into unified results
    # INPUTS: Checkpoint JSON files from all 128 models
    # OUTPUTS: Combined results list, Aggregated warnings list
    
    # Load all checkpoints and combine
    if verbose > 0:
        print("\nCombining checkpoint files...")
    
    all_results = []
    all_warnings = []
    
    for config in feature_combinations:
        model_id = config['id']
        checkpoint_path = checkpoints_dir / f"model_{model_id:03d}.json"
        
        if not checkpoint_path.exists():
            if verbose > 0:
                print(f"Warning: Checkpoint for model {model_id} not found, skipping")
            continue
        
        with open(checkpoint_path, 'r') as f:
            result = json.load(f)
        
        all_results.append(result)
        
        # Extract warnings if present (warnings are stored as strings)
        if 'warnings' in result and result['warnings']:
            for warning_str in result['warnings']:
                warning_dict = {
                    'model_id': model_id,
                    'model_name': result['model_name'],
                    'warning': warning_str
                }
                all_warnings.append(warning_dict)
    
    # Step 5: Save consolidated results
    # PURPOSE: Store evaluation results in organized directory structure
    # INPUTS: Combined results from all models, Warnings from failed evaluations
    # OUTPUTS: Results CSV file, Warnings CSV file (if warnings exist)
    # FUNCTION: results_manager.save_model_results
    
    # Create results DataFrame
    results_df = pd.DataFrame(all_results)
    
    # Save combined results through manager
    results_manager.save_model_results(results_df)
    
    if verbose > 0:
        print(f"Combined results saved through results manager")
    
    # Save warnings if any
    if all_warnings:
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        warnings_csv = warnings_dir / f"warnings_log_{timestamp}.csv"
        warnings_df = pd.DataFrame(all_warnings)
        warnings_df.to_csv(warnings_csv, index=False)
        
        if verbose > 0:
            print(f"Warnings saved to: {warnings_csv}")
            print(f"Total warnings: {len(all_warnings)}")
    
    # Step 6: Clean up temporary files
    # PURPOSE: Remove checkpoint files to conserve disk space
    # INPUTS: keep_checkpoints flag, Checkpoints directory
    # OUTPUTS: Deleted checkpoint directory (if keep_checkpoints=False)
    
    # Clean up checkpoints if requested
    if not keep_checkpoints:
        if verbose > 0:
            print("\nRemoving checkpoint files...")
        
        import shutil
        shutil.rmtree(checkpoints_dir)
        
        if verbose > 0:
            print("Checkpoints removed")
    
    return results_df

def _evaluate_single_feature_set(
    adata,
    y: np.ndarray,
    feature_set_config: Dict,
    param_grid: Dict,
    checkpoints_dir: Path,
    outer_cv_splits: int,
    outer_cv_repeats: int,
    inner_cv: int,
    random_state: int,
    verbose: bool,
    use_composite_features: bool = True,
    feature_mode: Optional[str] = None
) -> Dict:
    """
    Evaluate a single feature set and save checkpoint.
    
    This is a helper function called by evaluate_all_feature_sets in parallel.
    """
    # Sub-step 3.1: Configure feature set parameters
    # PURPOSE: Extract model identifiers and set feature engineering mode
    # INPUTS: Feature set configuration, Feature mode setting
    # OUTPUTS: Model ID, Model name, Configured feature_mode
    
    # Backwards compatibility logic
    if feature_mode is None:
        feature_mode = 'mixed' if use_composite_features else 'base'
        
    model_id = feature_set_config['id']
    model_name = feature_set_config['name']
    
    try:
        # Sub-step 3.2: Execute nested cross-validation
        # PURPOSE: Evaluate model performance using nested CV with hyperparameter tuning
        # INPUTS: Data (adata, y), Feature set config, Hyperparameter grid, CV settings
        # OUTPUTS: Cross-validation results with fold scores and best parameters
        # FUNCTION: nested_cv_single_feature_set
        
        # Run nested CV
        cv_results = nested_cv_single_feature_set(
            adata=adata,
            y=y,
            feature_set_config=feature_set_config,
            param_grid=param_grid,
            outer_cv_splits=outer_cv_splits,
            outer_cv_repeats=outer_cv_repeats,
            inner_cv=inner_cv,
            random_state=random_state,
            verbose=verbose,
            feature_mode=feature_mode
        )
        
        # Sub-step 3.3: Calculate performance statistics
        # PURPOSE: Compute standard error and standard deviation of AUROC across folds
        # INPUTS: Fold-level AUROC scores from nested CV
        # OUTPUTS: mean_auroc, se_auroc, sd_auroc
        
        # Calculate SE and SD
        fold_scores = cv_results['fold_scores']
        mean_auroc = cv_results['mean_auroc']
        n_folds = len(fold_scores)
        
        # Filter out NaN scores
        valid_scores = [s for s in fold_scores if not np.isnan(s)]
        
        if len(valid_scores) > 0:
            sd_auroc = np.std(valid_scores, ddof=1)
            se_auroc = sd_auroc / np.sqrt(len(valid_scores))
        else:
            sd_auroc = np.nan
            se_auroc = np.nan
        
        # Sub-step 3.4: Package and save results
        # PURPOSE: Create structured result dictionary and save checkpoint file
        # INPUTS: CV results, Performance statistics, Model metadata
        # OUTPUTS: Checkpoint JSON file for this model
        
        # Prepare result
        result = {
            'model_id': model_id,
            'model_name': model_name,
            'features': feature_set_config['features'],
            'n_base_features': len(feature_set_config['features']),
            'n_total_features': cv_results.get('n_features', np.nan),
            'mean_auroc': mean_auroc,
            'se_auroc': se_auroc,
            'sd_auroc': sd_auroc,
            'fold_scores': fold_scores,
            'fold_best_params': cv_results.get('fold_best_params', []),
            'coefficient_stats': cv_results.get('coefficient_stats', {}),
            'timestamp': datetime.now().isoformat()
        }
        
        # Extract warnings if present
        if 'warnings' in cv_results:
            result['warnings'] = cv_results['warnings']
        
        # Save checkpoint
        checkpoint_path = checkpoints_dir / f"model_{model_id:03d}.json"
        with open(checkpoint_path, 'w') as f:
            json.dump(result, f, indent=2)
        
        return result
        
    except Exception as e:
        # Log error and return failed result
        error_msg = f"Error evaluating model {model_id} ({model_name}): {str(e)}"
        warnings.warn(error_msg)
        
        result = {
            'model_id': model_id,
            'model_name': model_name,
            'features': feature_set_config['features'],
            'n_base_features': len(feature_set_config['features']),
            'n_total_features': np.nan,
            'mean_auroc': np.nan,
            'se_auroc': np.nan,
            'sd_auroc': np.nan,
            'fold_scores': [],
            'fold_best_params': [],
            'coefficient_stats': {},
            'timestamp': datetime.now().isoformat(),
            'error': error_msg
        }
        
        # Save checkpoint even for failed runs
        checkpoint_path = checkpoints_dir / f"model_{model_id:03d}.json"
        with open(checkpoint_path, 'w') as f:
            json.dump(result, f, indent=2)
        
        return result

def apply_1se_rule(
    results_df: pd.DataFrame,
    metric: str = 'mean_auroc',
    verbose: bool = True
) -> Dict:
    """
    Apply the 1-SE rule to select the simplest good model.
    
    The 1-SE rule selects the simplest model (fewest features) whose performance
    is within one standard error of the best model. This balances performance
    with interpretability and reduces overfitting risk.
    
    Parameters
    ----------
    results_df : pd.DataFrame
        Output from evaluate_all_feature_sets()
        Must have columns: model_id, model_name, n_total_features, mean_auroc, se_auroc
    metric : str
        Metric column name to use for selection (default 'mean_auroc')
    verbose : bool
        Print selection details and reasoning (default True)
        
    Returns
    -------
    selection_results : dict
        {
            'primary_model': dict (highest AUROC among tied simplest models),
            'all_candidates': list of dict (all models within 1-SE threshold),
            'tied_simplest': list of dict (all models tied for min n_features, if tie exists),
            'selection_reason': str (detailed explanation),
            'threshold': float (mean_auroc_best - se_best),
            'best_model': dict (model with max mean_auroc, for reference)
        }
    """
    # Ensure required columns exist
    required_cols = ['model_id', 'model_name', 'n_total_features', metric, 'se_auroc']
    missing_cols = [col for col in required_cols if col not in results_df.columns]
    if missing_cols:
        raise ValueError(f"Missing required columns: {missing_cols}")
    
    # Filter out models with NaN metric values
    valid_df = results_df[~results_df[metric].isna()].copy()
    
    if len(valid_df) == 0:
        raise ValueError("No valid models (all have NaN metric values)")
    
    # Find best model
    best_idx = valid_df[metric].idxmax()
    best_model = valid_df.loc[best_idx]
    
    best_auroc = best_model[metric]
    best_se = best_model['se_auroc']
    
    # Calculate threshold
    threshold = best_auroc - best_se
    
    # Filter candidates (models within 1-SE)
    candidates_df = valid_df[valid_df[metric] >= threshold].copy()
    
    # Find simplest model(s)
    min_features = candidates_df['n_total_features'].min()
    simplest_df = candidates_df[candidates_df['n_total_features'] == min_features].copy()
    
    # Sort by metric (descending) to get primary model
    simplest_df = simplest_df.sort_values(by=metric, ascending=False)
    
    # Primary model (highest AUROC among tied simplest)
    primary_model = simplest_df.iloc[0].to_dict()
    
    # All tied simplest models
    tied_simplest = [row.to_dict() for _, row in simplest_df.iterrows()]
    
    # All candidates within 1-SE
    all_candidates = [row.to_dict() for _, row in candidates_df.iterrows()]
    
    # Generate selection reason
    if len(tied_simplest) > 1:
        selection_reason = (
            f"Selected model {primary_model['model_id']} ({primary_model['model_name']}) "
            f"from {len(tied_simplest)} models tied for simplest ({min_features} features) "
            f"within 1-SE of best. Primary has highest AUROC: "
            f"{primary_model[metric]:.4f} ± {primary_model['se_auroc']:.4f}"
        )
    else:
        selection_reason = (
            f"Selected model {primary_model['model_id']} ({primary_model['model_name']}) "
            f"as simplest model ({min_features} features) within 1-SE of best. "
            f"AUROC: {primary_model[metric]:.4f} ± {primary_model['se_auroc']:.4f}"
        )
    
    # Create result dictionary
    selection_results = {
        'primary_model': primary_model,
        'all_candidates': all_candidates,
        'tied_simplest': tied_simplest,
        'selection_reason': selection_reason,
        'threshold': threshold,
        'best_model': best_model.to_dict()
    }
    
    # Print summary if verbose
    if verbose:
        print("\n" + "="*70)
        print("1-SE RULE MODEL SELECTION")
        print("="*70)
        print(f"\nBest Model (max {metric}):")
        print(f"  ID: {best_model['model_id']}")
        print(f"  Name: {best_model['model_name']}")
        print(f"  Features: {best_model['n_total_features']}")
        print(f"  {metric.upper()}: {best_auroc:.4f} ± {best_se:.4f}")
        
        print(f"\n1-SE Threshold: {threshold:.4f}")
        print(f"Models within 1-SE: {len(all_candidates)}")
        
        print(f"\nSimplest Model (min features):")
        print(f"  ID: {primary_model['model_id']}")
        print(f"  Name: {primary_model['model_name']}")
        print(f"  Features: {primary_model['n_total_features']}")
        print(f"  {metric.upper()}: {primary_model[metric]:.4f} ± {primary_model['se_auroc']:.4f}")
        
        if len(tied_simplest) > 1:
            print(f"\nNote: {len(tied_simplest)} models tied for simplest")
            print("  Other tied models:")
            for i, model in enumerate(tied_simplest[1:], 1):
                print(f"    {i}. Model {model['model_id']} ({model['model_name']}): "
                      f"{metric}={model[metric]:.4f}")
        
        print(f"\nSelection Reason:")
        print(f"  {selection_reason}")
        print("="*70 + "\n")
    
    return selection_results

def build_final_classifier(
    adata,
    y: np.ndarray,
    feature_set_config: Dict,
    param_grid: Dict,
    cv_folds: int = 10,
    random_state: int = 42,
    verbose: bool = True,
    use_composite_features: bool = True,
    feature_mode: Optional[str] = None
) -> Dict:
    """
    Build the final, deployable classifier trained on 100% of data.
    
    This function performs a complete training workflow:
    1. Re-run hyperparameter search on full dataset (no data held out)
    2. Train final model with optimal hyperparameters
    3. Find optimal classification threshold using out-of-fold predictions
    
    Parameters
    ----------
    adata : AnnData
        Single-cell data object with RAW protein intensities
    y : np.ndarray
        Binary labels (0/1)
    feature_set_config : dict
        Configuration for the selected feature set (from apply_1se_rule)
    param_grid : dict
        Hyperparameter grid for re-running GridSearchCV
    cv_folds : int
        Number of CV folds for threshold optimization (default 10)
        Uses StratifiedKFold (not Repeated) for out-of-fold predictions
    random_state : int
        Random seed for reproducibility
    verbose : bool
        Print progress information (default True)
    use_composite_features : bool
        Legacy parameter. Ignored if feature_mode is provided.
    feature_mode : str, optional
        'base', 'mixed', or 'composite'. If None, inferred from use_composite_features.
        
    Returns
    -------
    final_classifier : dict
        Complete deployable classifier package
    """
    from .nested_cv import _prepare_features
    
    # Backwards compatibility logic
    if feature_mode is None:
        feature_mode = 'mixed' if use_composite_features else 'base'
        
    if verbose:
        print("\n" + "="*70)
        print("BUILDING FINAL CLASSIFIER")
        print("="*70)
        print(f"\nFeature set: {feature_set_config['name']}")
        print(f"Base features: {len(feature_set_config['features'])}")

def build_calibrated_classifier(
    adata,
    y: np.ndarray,
    feature_set_config: Dict,
    param_grid: Dict,
    cv_folds: int = 5,
    random_state: int = 42,
    verbose: bool = True,
    use_composite_features: bool = True,
    feature_mode: Optional[str] = None
) -> Dict:
    """
    Build calibrated classifier with isotonic regression and valid OOF predictions.
    
    This function implements a 4-step workflow for building calibrated models:
    1. Grid Search with scoring='roc_auc' to find best hyperparameters
    2. Create CalibratedClassifierCV with isotonic method
    3. Generate out-of-fold predictions using cross_val_predict (for valid Brier scores)
    4. Fit final calibrated model on all data
    
    Parameters
    ----------
    adata : AnnData
        Single-cell data object with RAW protein intensities
    y : np.ndarray
        Binary labels (0/1)
    feature_set_config : dict
        Configuration for the selected feature set
    param_grid : dict
        Hyperparameter grid for GridSearchCV (e.g., {'C': [...], 'l1_ratio': [...]})
    cv_folds : int
        Number of CV folds for calibration and OOF predictions (default 5)
    random_state : int
        Random seed for reproducibility
    verbose : bool
        Print progress information (default True)
    use_composite_features : bool
        Legacy parameter. Ignored if feature_mode is provided.
    feature_mode : str, optional
        'base', 'mixed', or 'composite'. If None, inferred from use_composite_features.
        
    Returns
    -------
    calibrated_classifier : dict
        Complete deployable classifier package with:
        - 'model': Fitted CalibratedClassifierCV
        - 'scaler': Fitted StandardScaler
        - 'oof_probs': Out-of-fold probabilities (for valid Brier calculation)
        - 'valid_rows': Boolean mask of valid samples (after NaN removal)
        - 'feature_names': List of feature names
        - 'best_params': Best hyperparameters from grid search
        - 'is_calibrated': True flag
        - Other metadata
    """
    from .nested_cv import _prepare_features
    from sklearn.calibration import CalibratedClassifierCV
    
    # Backwards compatibility logic
    if feature_mode is None:
        feature_mode = 'mixed' if use_composite_features else 'base'
        
    if verbose:
        print("\n" + "="*70)
        print("BUILDING CALIBRATED CLASSIFIER")
        print("="*70)
        print(f"\nFeature set: {feature_set_config['name']}")
        print(f"Base features: {len(feature_set_config['features'])}")
        print(f"Feature mode: {feature_mode}")
        print(f"Calibration method: Isotonic regression")
    
    # Step 1: Prepare features on 100% of data
    if verbose:
        print("\nStep 1: Preparing features on full dataset...")
    
    # Extract selected proteins
    protein_list = feature_set_config['features']
    protein_indices = [list(adata.var_names).index(p) for p in protein_list]

    )
    
    # Remove samples with NaN
    y_clean = y[valid_rows]
    
    if verbose:
        print(f"  Total features (after NaN removal): {X_full_clean.shape[1]}")
        print(f"  Total samples (after NaN removal): {X_full_clean.shape[0]}")
    
    # Step 2: Find optimal hyperparameters on full dataset
    if verbose:
        print("\nStep 2: Finding optimal hyperparameters via GridSearchCV...")
    
    # Create Elastic Net classifier
    base_clf = LogisticRegression(
        penalty='elasticnet',
        solver='saga',
        class_weight='balanced',
        max_iter=10000,
        random_state=random_state
    )
    
    # Run GridSearchCV
    grid_search = GridSearchCV(
        estimator=base_clf,
        param_grid=param_grid,
        cv=5,
        scoring='roc_auc',
        n_jobs=-1,
        verbose=0
    )
    
    grid_search.fit(X_full_clean, y_clean)
    best_params = grid_search.best_params_
    
    if verbose:
        print(f"  Best hyperparameters: {best_params}")
        print(f"  Best CV AUROC: {grid_search.best_score_:.4f}")
    
    # Step 3: Create optimized model for calibration
    if verbose:

    optimized_model = LogisticRegression(

    calibrated_clf.fit(X_full_clean, y_clean)
    
    if verbose:

        'feature_set_config': feature_set_config,
        'best_params': best_params,
        'is_calibrated': True,

def find_optimal_threshold(
    y_true: np.ndarray,
    y_pred_proba: np.ndarray,
    metric: str = 'mcc',
    threshold_step: float = 0.01,
    verbose: bool = True
) -> Dict:
    """
    Find optimal classification threshold by linear sweep to maximize metric.
    
    Sweeps through threshold values from threshold_step to (1 - threshold_step)
    and selects the threshold that maximizes the specified metric.
    
    Parameters
    ----------
    y_true : np.ndarray, shape (n_samples,)
        True binary labels (0/1)
    y_pred_proba : np.ndarray, shape (n_samples,)
        Predicted probabilities from classifier (values in [0, 1])
    metric : str
        Metric to optimize (default 'mcc')
        Options: 'mcc' (Matthews Correlation Coefficient), 'f1' (F1 score)
    threshold_step : float
        Resolution for threshold sweep (default 0.01)
    verbose : bool
        Print optimal threshold and metric value (default True)
        
    Returns
    -------
    results : dict
        {
            'optimal_threshold': float,
            'max_metric_value': float,
            'all_thresholds': np.ndarray,
            'all_metric_values': np.ndarray,
            'metric_name': str
        }
    """
    # Generate thresholds to test
    thresholds = np.arange(threshold_step, 1.0, threshold_step)
    
    # Calculate metric for each threshold
    metric_values = []
    
    for threshold in thresholds:
        # Convert probabilities to binary predictions
        y_pred = (y_pred_proba >= threshold).astype(int)
        
        # Calculate metric
        if metric == 'mcc':
            value = matthews_corrcoef(y_true, y_pred)
        elif metric == 'f1':
            value = f1_score(y_true, y_pred)
        else:
            raise ValueError(f"Unknown metric: {metric}. Use 'mcc' or 'f1'")
        
        metric_values.append(value)
    
    metric_values = np.array(metric_values)
    
    # Find optimal threshold
    max_idx = np.argmax(metric_values)
    optimal_threshold = thresholds[max_idx]
    max_metric_value = metric_values[max_idx]
    
    # Print results if verbose
    if verbose:
        print(f"  Optimal threshold: {optimal_threshold:.3f}")
        print(f"  {metric.upper()} at threshold: {max_metric_value:.3f}")
    
    results = {
        'optimal_threshold': optimal_threshold,
        'max_metric_value': max_metric_value,
        'all_thresholds': thresholds,
        'all_metric_values': metric_values,
        'metric_name': metric
    }
    
    return results
