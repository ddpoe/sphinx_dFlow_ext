"""
Results Organization & Management for Elastic Net Analysis

This module provides tools for organizing and tracking Elastic Net analysis outputs.
Creates unique analysis identifiers and manages all results in structured directories.

Key Features:
=============
- Unique analysis IDs (never overwrite previous runs)
- Automatic directory structure creation
- Metadata tracking (who, what, when, how)
- Dual-format saving (CSV + PKL)
- Integrated plot management
- README generation for human readability

Typical Usage Flow:
===================

# Simple case - just cell line and condition
manager = ElasticNetResultsManager(
    cell_line="CFPAC",
    condition="control"
)
# → analysis_id: "CFPAC_control_20251113_142315"

# With additional info (e.g., cell cycle phase, treatment time)
manager = ElasticNetResultsManager(
    cell_line="HPAC",
    condition="ERK2000",
    additional_info="g0g1"
)
# → analysis_id: "HPAC_ERK2000_g0g1_20251113_150430"

# Double loop example (automated batch processing)
for cell_line in ['CFPAC', 'HPAC', 'BxPC3']:
    for condition in ['control', 'ERK2000', 'ERK5000']:
        # Optional: Add filtering info
        additional_info = "g0g1_filtered" if filter_2n else "allcells"
        
        manager = ElasticNetResultsManager(
            cell_line=cell_line,
            condition=condition,
            additional_info=additional_info
        )
        
        # Run full analysis...
        # All results automatically organized by manager

Classes:
========
- ElasticNetResultsManager: Main class for managing analysis outputs

Functions:
==========
- generate_analysis_id: Create unique identifier for analysis run
"""

def generate_analysis_id(
    cell_line: str,
    condition: str,
    additional_info: Optional[str] = None,
    timestamp: Optional[str] = None
) -> str:
    """
    Generate unique identifier for analysis run.
    
    Format: 
        - With additional_info: {cell_line}_{condition}_{additional_info}_{timestamp}
        - Without: {cell_line}_{condition}_{timestamp}
    
    Parameters
    ----------
    cell_line : str
        Cell line name (e.g., 'CFPAC', 'HPAC', 'Pa02C')
        Automatically converted to safe format (alphanumeric + underscores)
    condition : str
        Experimental condition (e.g., 'control', 'ERK2000', 'Fra1_KD')
        Automatically converted to safe format
    additional_info : str, optional
        Optional additional descriptor (e.g., 'g0g1', 'nonprolif', 'allcells', '48h')
        If None, omitted from ID
        Automatically converted to safe format
    timestamp : str, optional
        Timestamp in format YYYYMMDD_HHMMSS
        If None, auto-generated from current time
        
    Returns
    -------
    analysis_id : str
        Unique identifier, filesystem-safe
        
    Examples
    --------
    >>> # Minimal usage
    >>> generate_analysis_id("CFPAC", "control")
    'CFPAC_control_20251113_142315'
    
    >>> # With cell cycle phase info
    >>> generate_analysis_id("HPAC", "ERK2000", "g0g1")
    'HPAC_ERK2000_g0g1_20251113_150430'
    
    >>> # With treatment duration
    >>> generate_analysis_id("BxPC3", "Fra1_KD", "48h")
    'BxPC3_Fra1_KD_48h_20251113_160530'
    
    >>> # Reproducible ID (for testing)
    >>> generate_analysis_id("CFPAC", "control", "g0g1", "20251113_120000")
    'CFPAC_control_g0g1_20251113_120000'
    
    >>> # In a double loop
    >>> for cell_line in ['CFPAC', 'HPAC']:
    ...     for condition in ['control', 'ERK2000']:
    ...         aid = generate_analysis_id(cell_line, condition, "g0g1")
    ...         print(aid)
    'CFPAC_control_g0g1_20251113_142315'
    'CFPAC_ERK2000_g0g1_20251113_142316'
    'HPAC_control_g0g1_20251113_142317'
    'HPAC_ERK2000_g0g1_20251113_142318'
    
    Notes
    -----
    - IDs are filesystem-safe (spaces/special chars converted to underscores)
    - Timestamp ensures uniqueness even for same cell line/condition
    - additional_info is optional for flexibility
    - All components sanitized to prevent filesystem issues
    """
    # Sanitize inputs (replace spaces and special chars with underscores)
    def sanitize(s: str) -> str:
        """Convert string to filesystem-safe format."""
        import re
        # Replace anything that's not alphanumeric or underscore with underscore
        return re.sub(r'[^a-zA-Z0-9_]', '_', s)
    
    cell_line_safe = sanitize(cell_line)
    condition_safe = sanitize(condition)
    
    # Generate timestamp if not provided
    if timestamp is None:
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    
    # Build ID
    parts = [cell_line_safe, condition_safe]
    if additional_info is not None:
        parts.append(sanitize(additional_info))
    parts.append(timestamp)
    
    return "_".join(parts)

class ElasticNetResultsManager:
    """
    Manages all outputs for a single Elastic Net analysis run.
    
    Automatically organizes:
    - Model evaluation results (CSV + PKL)
    - Diagnostic plots (organized by type)
    - Final trained classifier
    - Preprocessing outputs (GMM histograms)
    - Warnings and logs
    - Metadata (who, what, when, how)
    
    Design Philosophy:
    ------------------
    - One manager = One complete analysis
    - Automatic directory creation
    - Prevents overwrites (unique IDs)
    - Tracks provenance (metadata)
    - Easy to use (minimal boilerplate)
    
    Directory Structure Created:
    ----------------------------
    results/
    └── {analysis_id}/
        ├── metadata.json              ← Analysis configuration
        ├── README.md                  ← Human-readable summary
        ├── all_models.csv             ← Excel-compatible results
        ├── all_models.pkl             ← Python-compatible (preserves types)
        ├── checkpoints/               ← Temporary (cleaned up after use)
        ├── warnings/
        │   └── warnings_log_{timestamp}.csv
        ├── plots/
        │   ├── model_selection.png
        │   ├── coefficient_stability.png
        │   └── ...
        ├── models/
        │   └── final_classifier.pkl
        └── preprocessing/
            ├── ploidy_gmm_histogram.png
            └── proliferation_gmm_histogram.png
    
    Examples
    --------
    >>> # Initialize manager (creates unique folder + metadata)
    >>> manager = ElasticNetResultsManager(
    ...     cell_line="CFPAC",
    ...     condition="control",
    ...     additional_info="g0g1"
    ... )
    >>> print(manager.analysis_id)
    'CFPAC_control_g0g1_20251113_142315'
    >>> 
    >>> # All outputs automatically organized
    >>> manager.save_model_results(results_df)  # Saves CSV + PKL
    >>> manager.save_plot(fig, "model_selection")  # Saves to plots/
    >>> manager.save_final_classifier(classifier)  # Saves to models/
    >>> 
    >>> # Generate human-readable summary
    >>> manager.create_summary()
    >>> 
    >>> # Double loop example (batch processing)
    >>> for cell_line in ['CFPAC', 'HPAC', 'BxPC3']:
    ...     for condition in ['control', 'ERK2000']:
    ...         manager = ElasticNetResultsManager(
    ...             cell_line=cell_line,
    ...             condition=condition,
    ...             additional_info="g0g1"
    ...         )
    ...         # Run analysis with this manager...
    ...         results_df = evaluate_all_feature_sets(
    ...             adata, y, feature_combos, param_grid,
    ...             results_manager=manager
    ...         )
    ...         manager.save_model_results(results_df)
    ...         # ...etc
    """

    def __init__(
        self,
        cell_line: str,
        condition: str,
        additional_info: Optional[str] = None,
        base_dir: str = "results",
        analysis_id: Optional[str] = None
    ):
        """
        Initialize results manager for a new analysis.
        
        Parameters
        ----------
        cell_line : str
            Cell line name (e.g., 'CFPAC', 'HPAC', 'Pa02C')
        condition : str
            Experimental condition (e.g., 'control', 'ERK2000', 'Fra1_KD')
        additional_info : str, optional
            Optional descriptor (e.g., 'g0g1', 'nonprolif', 'allcells', '48h')
            If None, omitted from analysis_id
        base_dir : str
            Base directory for all results (default: 'results')
        analysis_id : str, optional
            Use specific analysis ID (for loading existing results)
            If None, generates new ID with current timestamp
            
        Side Effects
        ------------
        - Creates directory structure under {base_dir}/{analysis_id}/
        - Saves initial metadata.json
        
        Attributes
        ----------
        analysis_id : str
            Unique identifier for this analysis
        cell_line : str
            Cell line name (stored for metadata)
        condition : str
            Experimental condition (stored for metadata)
        additional_info : str or None
            Additional descriptor (stored for metadata)
        base_dir : Path
            Root directory for all results
        analysis_dir : Path
            Full path to this analysis folder
        dirs : dict
            Paths to all subdirectories:
            - 'checkpoints': For evaluate_all_feature_sets()
            - 'warnings': For warning logs
            - 'plots': For diagnostic plots
            - 'models': For final classifier
            - 'preprocessing': For GMM histograms
        metadata : dict
            Analysis configuration and parameters
        """
        self.cell_line = cell_line
        self.condition = condition
        self.additional_info = additional_info
        self.base_dir = Path(base_dir)
        
        # Generate or use provided analysis ID
        if analysis_id is None:
            self.analysis_id = generate_analysis_id(cell_line, condition, additional_info)
        else:
            self.analysis_id = analysis_id
        
        # Set up directory structure
        self.analysis_dir = self.base_dir / self.analysis_id
        self.dirs = {
            'checkpoints': self.analysis_dir / 'checkpoints',
            'warnings': self.analysis_dir / 'warnings',
            'plots': self.analysis_dir / 'plots',
            'models': self.analysis_dir / 'models',
            'preprocessing': self.analysis_dir / 'preprocessing'
        }
        
        # Create all directories
        self.analysis_dir.mkdir(parents=True, exist_ok=True)
        for dir_path in self.dirs.values():
            dir_path.mkdir(parents=True, exist_ok=True)
        
        # Initialize metadata
        self.metadata = {
            'analysis_id': self.analysis_id,
            'cell_line': self.cell_line,
            'condition': self.condition,
            'additional_info': self.additional_info,
            'timestamp': datetime.now().isoformat(),
            'created_at': datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }
        
        # Save initial metadata
        self.save_metadata()

    def save_metadata(self, **kwargs):
        """
        Save or update analysis metadata.
        
        Parameters
        ----------
        **kwargs : dict
            Arbitrary metadata to save/update
            Common keys:
            - data_file: str (path to input data)
            - n_cells_original: int
            - n_cells_final: int
            - filter_2n: bool
            - fra1_filter_config: dict
            - hyperparameters: dict
            - cv_config: dict
            - selected_model: dict
            
        Saves to:
        ---------
        {analysis_dir}/metadata.json
        
        Example Metadata Structure:
        ---------------------------
        {
            "analysis_id": "CFPAC_control_g0g1_20251113_142315",
            "cell_line": "CFPAC",
            "condition": "control",
            "additional_info": "g0g1",
            "timestamp": "2025-11-13T14:23:15",
            "data_file": "/path/to/data.h5ad",
            "n_cells_original": 50000,
            "n_cells_final": 32000,
            "filter_2n": true,
            "fra1_filter": {
                "enabled": true,
                "percentile": 90,
                "reference": "nonprolif"
            },
            "hyperparameters": {"C": [...], "l1_ratio": [...]},
            "cv_config": {"outer_splits": 10, ...},
            "selected_model": {
                "model_id": 42,
                "model_name": "CycD1_CDK4_p27",
                "mean_auroc": 0.941
            }
        }
        
        Notes
        -----
        - Called automatically on __init__
        - Can be called multiple times to update metadata
        - Updates are merged with existing metadata
        """
        # Update metadata with new kwargs
        self.metadata.update(kwargs)
        
        # Add update timestamp
        self.metadata['last_updated'] = datetime.now().isoformat()
        
        # Save to file
        metadata_path = self.analysis_dir / 'metadata.json'
        with open(metadata_path, 'w') as f:
            json.dump(self.metadata, f, indent=2, default=str)

    def save_model_results(self, results_df: pd.DataFrame):
        """
        Save model evaluation results in dual format.
        
        Parameters
        ----------
        results_df : pd.DataFrame
            Output from evaluate_all_feature_sets()
            
        Saves to:
        ---------
        - {analysis_dir}/all_models.csv  # Human-readable, Excel-compatible
        - {analysis_dir}/all_models.pkl  # Preserves Python types (lists, dicts)
        
        Why Both Formats?
        -----------------
        - CSV: Easy to open in Excel, share with collaborators
        - PKL: Preserves list/dict types, can reload for further analysis
        
        Example
        -------
        >>> results_df = evaluate_all_feature_sets(...)
        >>> manager.save_model_results(results_df)
        >>> # Later, reload:
        >>> results_df = manager.load_model_results()
        """
        csv_path = self.analysis_dir / 'all_models.csv'
        pkl_path = self.analysis_dir / 'all_models.pkl'
        
        # Save CSV (human-readable)
        results_df.to_csv(csv_path, index=False)
        
        # Save PKL (preserves types)
        results_df.to_pickle(pkl_path)
        
        # Update metadata
        self.save_metadata(
            n_models_evaluated=len(results_df),
            results_saved_at=datetime.now().isoformat()
        )

    def load_model_results(self) -> pd.DataFrame:
        """
        Load previously saved model results.
        
        Returns
        -------
        results_df : pd.DataFrame
            Full results with Python types preserved (from PKL)
            
        Raises
        ------
        FileNotFoundError
            If all_models.pkl doesn't exist
            
        Example
        -------
        >>> # Load existing analysis
        >>> manager = ElasticNetResultsManager.load("CFPAC_control_g0g1_20251113_142315")
        >>> results_df = manager.load_model_results()
        """
        pkl_path = self.analysis_dir / 'all_models.pkl'
        if not pkl_path.exists():
            raise FileNotFoundError(f"No results file found at {pkl_path}")
        return pd.read_pickle(pkl_path)

    def save_plot(
        self,
        fig: 'Figure',
        name: str,
        dpi: int = 150,
        bbox_inches: str = 'tight'
    ) -> Path:
        """
        Save plot with automatic naming and organization.
        
        Parameters
        ----------
        fig : matplotlib.Figure
            Figure to save
        name : str
            Base name (without extension)
            Examples: 'model_selection', 'coefficient_stability', 'threshold_sweep'
        dpi : int
            Resolution (default 150 for publication quality)
        bbox_inches : str
            Bounding box (default 'tight' to avoid clipping)
            
        Returns
        -------
        save_path : Path
            Full path to saved plot
            
        Saves to:
        ---------
        {analysis_dir}/plots/{name}.png
        
        Example
        -------
        >>> fig, ax = plot_model_selection(results_df, selection)
        >>> manager.save_plot(fig, 'model_selection')
        PosixPath('.../CFPAC_control_g0g1_20251113_142315/plots/model_selection.png')
        """
        save_path = self.dirs['plots'] / f"{name}.png"
        fig.savefig(save_path, dpi=dpi, bbox_inches=bbox_inches)
        plt.close(fig)  # Free memory
        return save_path

    def save_final_classifier(self, classifier_dict: dict) -> Path:
        """
        Save final trained classifier for deployment.
        
        Parameters
        ----------
        classifier_dict : dict
            Output from build_final_classifier()
            Contains: model, scaler, feature_names, optimal_threshold, etc.
            
        Returns
        -------
        save_path : Path
            Full path to saved classifier
            
        Saves to:
        ---------
        {analysis_dir}/models/final_classifier.pkl
        
        Example
        -------
        >>> classifier = build_final_classifier(...)
        >>> manager.save_final_classifier(classifier)
        >>> # Later, reload:
        >>> classifier = manager.load_final_classifier()
        """
        save_path = self.dirs['models'] / 'final_classifier.pkl'
        joblib.dump(classifier_dict, save_path)
        
        # Update metadata with model info
        self.save_metadata(
            final_classifier_saved=True,
            classifier_saved_at=datetime.now().isoformat()
        )
        
        return save_path

    @classmethod
    def load(cls, analysis_id: str, base_dir: str = "results"):
        """
        Load existing analysis results.
        
        Parameters
        ----------
        analysis_id : str
            ID of analysis to load (full folder name)
        base_dir : str
            Base results directory (default: 'results')
            
        Returns
        -------
        manager : ElasticNetResultsManager
            Manager instance with access to all saved results
            
        Raises
        ------
        FileNotFoundError
            If analysis directory or metadata.json doesn't exist
            
        Example
        -------
        >>> # Load previous analysis
        >>> manager = ElasticNetResultsManager.load(
        ...     "CFPAC_control_g0g1_20251113_142315"
        ... )
        >>> results_df = manager.load_model_results()
        >>> classifier = manager.load_final_classifier()
        >>> print(manager.metadata)
        """
        analysis_dir = Path(base_dir) / analysis_id
        if not analysis_dir.exists():
            raise FileNotFoundError(f"Analysis directory not found: {analysis_dir}")
        
        metadata_path = analysis_dir / 'metadata.json'
        if not metadata_path.exists():
            raise FileNotFoundError(f"Metadata file not found: {metadata_path}")
        
        # Load metadata
        with open(metadata_path, 'r') as f:
            metadata = json.load(f)
        
        # Create manager instance
        manager = cls(
            cell_line=metadata.get('cell_line', 'Unknown'),
            condition=metadata.get('condition', 'Unknown'),
            additional_info=metadata.get('additional_info'),
            base_dir=base_dir,
            analysis_id=analysis_id
        )
        
        # Restore full metadata
        manager.metadata = metadata
        
        return manager
