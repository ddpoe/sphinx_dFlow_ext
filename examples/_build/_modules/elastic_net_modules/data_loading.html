class AdataProcessor:
    """Processes AnnData objects for cell cycle elastic net analysis.

    This class loads a base AnnData object for a given cell line and provides
    methods to sequentially add different data layers required for analysis,
    such as dimensionality reduction coordinates and cell cycle phase labels.

    Attributes:
        cell_line: The name of the cell line being processed.
        cell_info: A dictionary containing configuration and parameters for the
            cell line.
        adata: The primary AnnData object for the cell line.
        phate_coords_obsm_key: The key for PHATE coordinates in `.obsm`.
        phate_params_uns_key: The key for PHATE parameters in `.uns`.
        phase_labels_obs_key: The key for phase labels in `.obs`.
        phase_label_colors_obs_key: The key for phase label colors in `.obs`.
    """

    def __init__(
        self,
        cell_line: str,
        optimal_cell_lines_dict: Dict[str, Any],
        pdac_data_dir: Path,
        cc_mapping_package_dir: Path,
        adata_filename_suffix: str = '_GMM_DIP.h5ad',
        phate_coords_obsm_key: str = 'PhateCoords',
        phate_params_uns_key: str = 'phate_coords_args',
        phase_labels_obs_key: str = 'PhaseLabels',
        phase_label_colors_obs_key: str = 'PhaseLabelColors',
        feature_set_var_key: str = 'in_random_forest_fs',
        feature_set_uns_key: str = 'random_forest_fs',
        random_forest_fs_name_key: str = 'RandomForestFeatureSets',
        random_forest_fs_key: str = 'RandomForestFeatureSet',
    ):
        """Initializes the AdataProcessor.

        Args:
            cell_line: The name of the cell line.
            optimal_cell_lines_dict: A dictionary containing configuration for
                all cell lines.
            pdac_data_dir: Path to the PDAC data directory.
            cc_mapping_package_dir: Path to the cc_mapping package directory.
            adata_filename_suffix: Suffix for the AnnData filename.
            phate_coords_obsm_key: Key for PHATE coordinates in `.obsm`.
            phate_params_uns_key: Key for PHATE parameters in `.uns`.
            phase_labels_obs_key: Key for phase labels in `.obs`.
            phase_label_colors_obs_key: Key for phase label colors in `.obs`.
            feature_set_var_key: Key for the feature set boolean mask in `.var`.
            feature_set_uns_key: Key for the feature set list in `.uns`.
            random_forest_fs_name_key: The key for the RF feature set name in
                the API results.
            random_forest_fs_key: The key for the actual feature set in the
                API results.
        """
        self.cell_line = cell_line
        self.pdac_data_dir = pdac_data_dir
        self.cc_mapping_package_dir = cc_mapping_package_dir
        self.random_forest_fs_name_key = random_forest_fs_name_key
        self.random_forest_fs_key = random_forest_fs_key
        self.cell_info = optimal_cell_lines_dict[cell_line]

        # Define default keys as instance attributes
        self.adata_filename_suffix = adata_filename_suffix
        self.phate_coords_obsm_key = phate_coords_obsm_key
        self.phate_params_uns_key = phate_params_uns_key
        self.phase_labels_obs_key = phase_labels_obs_key
        self.phase_label_colors_obs_key = phase_label_colors_obs_key
        self.feature_set_var_key = feature_set_var_key
        self.feature_set_uns_key = feature_set_uns_key

        # Import cc_mapping_local API from local directory (separate from PyPI package)
        # The local version is renamed to avoid conflicts with the installed cc-mapping package
        
        # Add the PARENT directory of cc_mapping_local to sys.path
        # so Python can find the cc_mapping_local package
        parent_dir = cc_mapping_package_dir.parent
        sys.path.insert(0, str(parent_dir))
        
        # Now import from local version (renamed to cc_mapping_local)
        from cc_mapping_local import api, preprocess

        self.api = api
        self.preprocess = preprocess
        
        # Remove the parent path after import to keep sys.path clean
        sys.path.remove(str(parent_dir))

        self.adata = self._load_adata()



    def _load_adata(self) -> ad.AnnData:
        """Loads the base AnnData object from its H5AD file.

        Returns:
            The loaded AnnData object.

        Raises:
            FileNotFoundError: If the H5AD file for the cell line does not
                exist at the expected path.
        """
        cell_line_path = (
            self.pdac_data_dir / self.cell_line / f'{self.cell_line}_GMM_DIP.h5ad'
        )
        if not cell_line_path.exists():
            raise FileNotFoundError(f"AnnData file not found at: {cell_line_path}")
        return ad.read_h5ad(cell_line_path)

    def _get_random_forest_fs_name(self) -> str:
        """Constructs the name for the random forest feature set.

        This name is used to identify the correct files for PHATE results
        that were generated using a specific feature set.

        Returns:
            The formatted name string for the random forest feature set.
        """
        sample_name = self.cell_line
        g0mo_root_name = self.cell_info['G0MORootName']
        rf_fs_root_name = self.cell_info['RandomForestFSRootName']

        if rf_fs_root_name:
            return f'{sample_name}_{g0mo_root_name}_{rf_fs_root_name}'
        return f'{sample_name}_{g0mo_root_name}'

    def _get_g0mo_name(self) -> str:
        """Constructs the name for the G0/M-phase optimized results.

        Returns:
            The formatted name string for the G0MO results.
        """
        return f"{self.cell_line}_{self.cell_info['G0MORootName']}"

    def add_random_forest_feature_set(
        self,
        feature_set_var_key: Optional[str] = None,
        feature_set_uns_key: Optional[str] = None,
    ):
        """Adds the random forest feature set to .var and .uns.

        This method retrieves the list of features from the random forest
        feature selection, adds a boolean mask to `.var` indicating which
        features are in the set, and stores the list of feature names in `.uns`.

        Args:
            feature_set_var_key: The key in `.var` for the boolean mask.
                If None, `self.feature_set_var_key` is used.
            feature_set_uns_key: The key in `.uns` for the list of features.
                If None, `self.feature_set_uns_key` is used.
        """
        var_key = feature_set_var_key or self.feature_set_var_key
        uns_key = feature_set_uns_key or self.feature_set_uns_key

        rf_fs_name = self._get_random_forest_fs_name()
        rf_results = self.api.GetRandomForestFSResults(
            RandomForestFSName=rf_fs_name, SampleName=self.cell_line
        )

        # We expect 'RandomForestFeatureSet' to be a key in the returned dictionary.
        # If not, this will raise a KeyError, which is appropriate.
        feature_set = rf_results[self.random_forest_fs_name_key]

        self.adata.var[var_key] = self.adata.var_names.isin(feature_set)
        self.adata.uns[uns_key] = feature_set

    def add_phate_coordinates(
        self,
        phate_coords_obsm_key: Optional[str] = None,
        phate_params_uns_key: Optional[str] = None,
    ):
        """Computes and adds PHATE coordinates to the AnnData object.

        This method retrieves pre-computed PHATE coordinates, which are based on
        a random forest feature set, and adds them to the `.obsm` attribute of
        the main AnnData object.

        Args:
            phate_coords_obsm_key: The key in `.obsm` where PHATE coordinates
                will be stored. If None, `self.phate_coords_obsm_key` is used.
            phate_params_uns_key: The key in `.uns` where PHATE parameters
                will be stored. If None, `self.phate_params_uns_key` is used.
        """
        obsm_key = phate_coords_obsm_key or self.phate_coords_obsm_key
        params_key = phate_params_uns_key or self.phate_params_uns_key
        rf_fs_name = self._get_random_forest_fs_name()
        phate_results = self.api.ReturnPhateResults(
            SampleName=self.cell_line,
            RandomForestFSName=rf_fs_name,
            PhateArgs=self.cell_info['PhateParameters'],
        )
        phate_coords_file = phate_results['PhateCoordsSavePath']
        phate_coords = pd.read_csv(phate_coords_file, index_col=0)
        phate_coords.index = self.adata.obs.index
        self.adata.obsm[obsm_key] = phate_coords.iloc[:, 1:].values
        self.adata.uns[params_key] = self.cell_info['PhateParameters']

    def add_phase_labels(
        self,
        phase_labels_obs_key: Optional[str] = None,
        phase_label_colors_obs_key: Optional[str] = None,
    ):
        """Adds cell cycle phase labels and colors to the AnnData object.

        This method retrieves pre-computed phase labels and their corresponding
        color mappings and adds them to the `.obs` attribute of the main
        AnnData object.

        Args:
            phase_labels_obs_key: The key in `.obs` for phase labels.
                If None, `self.phase_labels_obs_key` is used.
            phase_label_colors_obs_key: The key in `.obs` for phase label
                colors. If None, `self.phase_label_colors_obs_key` is used.
        """
        labels_key = phase_labels_obs_key or self.phase_labels_obs_key
        colors_key = phase_label_colors_obs_key or self.phase_label_colors_obs_key
        g0mo_name = self._get_g0mo_name()
        g0mo_results = self.api.get_G0MO3_results(
            G0MOName=g0mo_name, SampleName=self.cell_line
        )
        phase_label_path = g0mo_results['PhaseLabelSavePath']

        phase_labels = pd.read_csv(phase_label_path, index_col=0).iloc[:, 0]
        phase_color_mappings = self.adata.uns['cc_phase_color_mappings']
        phase_colors = phase_labels.map(phase_color_mappings)

        self.adata.obs[labels_key] = phase_labels.values
        self.adata.obs[colors_key] = phase_colors.values

    def process_all(self) -> Tuple[ad.AnnData, None]:
        """Runs all data processing steps in the correct order.

        This convenience method calls all the individual data-adding methods
        to fully process the AnnData object (without Palantir results).

        Returns:
            A tuple containing:
            - The main, enriched AnnData object.
            - None (placeholder for compatibility with original API).
        """
        self.add_random_forest_feature_set()
        self.add_phate_coordinates()
        self.add_phase_labels()
        return self.adata, None
