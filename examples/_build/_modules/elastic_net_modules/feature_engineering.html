"""
Feature Engineering Module for Elastic Net Analysis

This module handles:
1. Log transformation of protein intensities (for exploration/validation)
2. Generation of all 2^N feature combinations with validation and case-insensitive matching
3. Dynamic computation of 6 composite features for each combination:
   - SUM_GAS: Sum of activator proteins (cycD1, cycE) - NaN if < 2 present
   - SUM_BRAKE: Sum of inhibitor proteins (p21, p27) - NaN if < 2 present
   - SUM_ENGINES: Sum of engine proteins (CDK2, CDK4, CDK6) - NaN if < 2 present
   - GAS_BRAKE_RATIO: SUM_GAS - SUM_BRAKE (log-space difference) - NaN if either is NaN
   - SUM_ACTIVATED_ENGINE: Sum of biologically valid cyclin-CDK products - NaN if < 1 complex
   - ACT_ENGINE_BRAKE_RATIO: SUM_ACTIVATED_ENGINE - BRAKE - NaN if no complexes or brakes

Key Design Principles:
- log_transform_features(): Available for exploratory analysis and validation notebooks
- In nested CV (Phase 1.2): Log transform happens INSIDE the CV loop to prevent data leakage
- generate_all_feature_combinations(): Case-insensitive validation, preserves user's naming convention
- compute_composite_features(): Expects Z-SCORED data as input (mean≈0, std≈1)
- Z-scoring is done per fold in nested CV (train scaler fit on train data only)
- NaN handling: Composites return NaN when < 2 components (not meaningful sums)
"""

import numpy as np
import pandas as pd
from itertools import product
from typing import List, Dict, Tuple, Optional
import warnings

def log_transform_features(adata, protein_list: List[str], 
                           layer_name: str = 'log_transformed',
                           pseudocount: float = 1e-100) -> None:
    """
    Apply log10 transformation to raw protein intensities.
    
    This is the first step in feature engineering. Raw protein intensities
    are typically highly skewed and need log transformation for:
    1. Normalization of distributions
    2. Stabilization of variance
    3. Biological interpretability (fold changes become linear)
    
    Parameters
    ----------
    adata : AnnData
        Single-cell data object with raw protein intensities in .X or a layer
    protein_list : list of str
        List of protein names to transform (e.g., ['CycD1', 'CycE', 'p21', ...])
    layer_name : str, optional
        Name for the new layer to store log-transformed data (default: 'log_transformed')
    pseudocount : float, optional
        Minimal safety value added before log transform (default: 1e-100)
        Data should already be filtered to remove zeros; this is just protection
        
    Returns
    -------
    None
        Modifies adata in-place by adding a new layer
        
    Notes
    -----
    - Data should already be filtered to remove zeros before calling this function
    - Minimal pseudocount (1e-100) added as safety net without distorting data
    - Uses log10 (base 10) for interpretability
    - Stores result in adata.layers[layer_name]
    - Original data in adata.X is NOT modified
    
    Examples
    --------
    >>> import anndata as ad
    >>> import numpy as np
    >>> # Create mock data
    >>> adata = ad.AnnData(X=np.random.lognormal(0, 1, (100, 7)))
    >>> adata.var_names = ['CycD1', 'CycE', 'p21', 'p27', 'CDK2', 'CDK4', 'CDK6']
    >>> log_transform_features(adata, adata.var_names.tolist())
    >>> print(adata.layers.keys())  # dict_keys(['log_transformed'])
    """
    # Extract raw data for specified proteins
    if isinstance(adata.X, pd.DataFrame):
        X_raw = adata[:, protein_list].X.values
    else:
        protein_indices = [list(adata.var_names).index(p) for p in protein_list]
        X_raw = adata.X[:, protein_indices]
    
    # Apply log10 transformation with minimal pseudocount for safety
    with warnings.catch_warnings():
        warnings.filterwarnings('ignore', category=RuntimeWarning)
        X_log = np.log10(X_raw + pseudocount)
    
    # Create full array to store in layer (must match adata.X shape)
    X_full = adata.X.copy() if hasattr(adata.X, 'copy') else np.array(adata.X)
    protein_indices = [list(adata.var_names).index(p) for p in protein_list]
    for i, idx in enumerate(protein_indices):
        X_full[:, idx] = X_log[:, i]
    
    # Store in new layer
    adata.layers[layer_name] = X_full
    
    # Store metadata about transformation
    if 'log_transform_params' not in adata.uns:
        adata.uns['log_transform_params'] = {}
    adata.uns['log_transform_params'] = {
        'proteins': protein_list,
        'pseudocount': pseudocount,
        'layer_name': layer_name
    }

def generate_all_feature_combinations(
    activators: List[str],
    inhibitors: List[str],
    engines: List[str]
) -> List[Dict]:
    """
    Generate all 2^N combinations of activators, inhibitors, and engines.
    
    This creates every possible subset of the N cell cycle proteins, ranging from
    the empty set (0 proteins) to the full set (all N proteins). Each combination
    becomes a candidate feature set for model evaluation.
    
    Parameters
    ----------
    activators : list of str
        List of activator proteins (GAS), e.g., ['cycE', 'cycD1'].
        Must contain only cyclins: CycD1/cycD1 or CycE/cycE (case-insensitive).
        Pass empty list [] to skip validation (for future extensibility).
    inhibitors : list of str
        List of inhibitor proteins (BRAKE), e.g., ['p21', 'p27'].
        Must contain only known inhibitors: p21, p27 (case-insensitive).
        Pass empty list [] to skip validation (for future extensibility).
    engines : list of str
        List of engine proteins (CDKs), e.g., ['CDK2', 'CDK4', 'CDK6'].
        Must contain only known CDKs: CDK2, CDK4, CDK6 (case-insensitive).
        Pass empty list [] to skip validation (for future extensibility).
        
    Returns
    -------
    combinations : list of dict
        Each dict contains:
        - 'features': list of protein names in this combination
        - 'gas': subset of activators in this combination
        - 'brakes': subset of inhibitors in this combination
        - 'engine': subset of engine proteins in this combination
        - 'name': human-readable name (e.g., "cycD1_CDK4_p27")
        - 'id': unique integer ID (0 to 2^N-1)
        - 'n_features': number of base proteins (before adding composites)
        - 'valid_complexes': list of tuples describing valid cyclin-CDK pairs
          Example: [('cycD1', ['CDK4', 'CDK6']), ('cycE', ['CDK2'])]
          
    Raises
    ------
    ValueError
        If any protein name doesn't match expected cell cycle proteins (case-insensitive).
        Empty lists are allowed and skip validation.
          
    Notes
    -----
    Expected proteins (case-insensitive):
    - GAS (activators): CycD1, CycE
    - BRAKE (inhibitors): p21, p27
    - ENGINE (catalysts): CDK2, CDK4, CDK6
    
    Biologically Valid Cyclin-CDK Complexes:
    - ✅ CycD1-CDK4 (G1 phase progression)
    - ✅ CycD1-CDK6 (G1 phase progression, alternative to CDK4)
    - ✅ CycE-CDK2 (G1/S transition)
    
    Invalid pairings (NOT computed):
    - ❌ CycD1-CDK2 (not a natural complex)
    - ❌ CycE-CDK4 (not a natural complex)
    - ❌ CycE-CDK6 (not a natural complex)
    
    Examples
    --------
    >>> # Standard usage with lowercase names
    >>> combos = generate_all_feature_combinations(
    ...     activators=['cycE', 'cycD1'],
    ...     inhibitors=['p21', 'p27'],
    ...     engines=['CDK2', 'CDK4', 'CDK6']
    ... )
    >>> len(combos)
    128
    >>> # Empty set (model 0)
    >>> combos[0]
    {'features': [], 'gas': [], 'brakes': [], 'engine': [], 'name': 'empty', 'id': 0, ...}
    >>> # Full set (model 127)
    >>> combos[127]['features']
    ['cycE', 'cycD1', 'p21', 'p27', 'CDK2', 'CDK4', 'CDK6']
    >>> # Example with valid complexes (case preserved from input)
    >>> combo = [c for c in combos if set(c['features']) == {'cycD1', 'CDK4', 'p27'}][0]
    >>> combo['valid_complexes']
    [('cycD1', ['CDK4'])]
    """
    # Define expected proteins (case-insensitive validation)
    EXPECTED_CYCLINS = {'cycd1', 'cyce'}
    EXPECTED_INHIBITORS = {'p21', 'p27'}
    EXPECTED_CDKS = {'cdk2', 'cdk4', 'cdk6'}
    
    # Validate activators (only if non-empty)
    if activators:
        invalid = [p for p in activators if p.lower() not in EXPECTED_CYCLINS]
        if invalid:
            raise ValueError(
                f"Invalid activator proteins: {invalid}. "
                f"Expected (case-insensitive): CycD1 or CycE. "
                f"Pass empty list [] to skip validation."
            )
    
    # Validate inhibitors (only if non-empty)
    if inhibitors:
        invalid = [p for p in inhibitors if p.lower() not in EXPECTED_INHIBITORS]
        if invalid:
            raise ValueError(
                f"Invalid inhibitor proteins: {invalid}. "
                f"Expected (case-insensitive): p21, p27. "
                f"Pass empty list [] to skip validation."
            )
    
    # Validate engines (only if non-empty)
    if engines:
        invalid = [p for p in engines if p.lower() not in EXPECTED_CDKS]
        if invalid:
            raise ValueError(
                f"Invalid engine proteins: {invalid}. "
                f"Expected (case-insensitive): CDK2, CDK4, CDK6. "
                f"Pass empty list [] to skip validation."
            )
    
    # Helper function for case-insensitive protein matching
    def _is_protein(name: str, target: str) -> bool:
        """Check if protein name matches target (case-insensitive)"""
        return name.lower() == target.lower()
    
    # All proteins
    all_proteins = activators + inhibitors + engines
    n_proteins = len(all_proteins)
    
    # Generate all binary combinations (2^n)
    combinations = []
    for i, combo_bits in enumerate(product([0, 1], repeat=n_proteins)):
        # Select proteins where bit is 1
        selected_proteins = [p for p, bit in zip(all_proteins, combo_bits) if bit == 1]
        
        # Categorize proteins
        gas_in_set = [p for p in selected_proteins if p in activators]
        brakes_in_set = [p for p in selected_proteins if p in inhibitors]
        engine_in_set = [p for p in selected_proteins if p in engines]
        
        # Determine valid cyclin-CDK complexes for this combination (case-insensitive)
        valid_complexes = []
        
        # Check CycD1 pathway: CycD1 can pair with CDK4 and/or CDK6
        cycd1_proteins = [p for p in gas_in_set if _is_protein(p, 'CycD1')]
        if cycd1_proteins:
            cycd1 = cycd1_proteins[0]  # Use actual case from user input
            cycd1_partners = [cdk for cdk in engine_in_set 
                             if _is_protein(cdk, 'CDK4') or _is_protein(cdk, 'CDK6')]
            if cycd1_partners:
                valid_complexes.append((cycd1, cycd1_partners))
        
        # Check CycE pathway: CycE pairs ONLY with CDK2
        cyce_proteins = [p for p in gas_in_set if _is_protein(p, 'CycE')]
        cdk2_proteins = [p for p in engine_in_set if _is_protein(p, 'CDK2')]
        if cyce_proteins and cdk2_proteins:
            valid_complexes.append((cyce_proteins[0], [cdk2_proteins[0]]))
        
        # Create human-readable name
        if not selected_proteins:
            name = 'empty'
        else:
            name = '_'.join(selected_proteins)
        
        # Store combination metadata
        combinations.append({
            'id': i,
            'name': name,
            'features': selected_proteins,
            'n_features': len(selected_proteins),
            'gas': gas_in_set,
            'brakes': brakes_in_set,
            'engine': engine_in_set,
            'valid_complexes': valid_complexes
        })
    
    return combinations

def logsumexp10(values):
    """
    Calculate log10(sum(10^values)) for LogSumExp in base 10.
    
    This helper converts Z-scores (which represent log10 abundances) back to
    linear space, sums them, then converts back to log10 space.
    
    Mathematical relationship:
    - If Z = log10(abundance), then abundance = 10^Z
    - Sum of abundances = 10^Z1 + 10^Z2 + ...
    - Log10 of sum = log10(10^Z1 + 10^Z2 + ...)
    
    Parameters
    ----------
    values : list or array-like
        List of values in log10 space (Z-scores)
        
    Returns
    -------
    result : float or np.ndarray
        log10(sum(10^values))
        
    Notes
    -----
    Uses np.logaddexp for numerical stability to avoid overflow/underflow.
    Works with both single values and arrays of values.
    
    Examples
    --------
    >>> logsumexp10([1.0, 2.0])  # log10(10 + 100) = log10(110) ≈ 2.041
    2.0413926851582251
    >>> logsumexp10([0.0, 0.0])  # log10(1 + 1) = log10(2) ≈ 0.301
    0.3010299956639812
    """
    values = np.asarray(values)
    if values.size == 0:
        return np.nan
    
    ln10 = np.log(10)
    if values.ndim == 1:
        # Single list of values
        return np.logaddexp.reduce(values * ln10) / ln10
    else:
        # Multiple rows (for cell-wise computation)
        return np.logaddexp.reduce(values * ln10, axis=1) / ln10

def compute_composite_features(
    X_base: np.ndarray,
    feature_set_config: Dict,
    feature_names: List[str]
) -> Tuple[np.ndarray, List[str]]:
    """
    Compute GAS, BRAKE, ENGINES, GAS_BRAKE_RATIO, ACTIVATED_ENGINE, and ACT_ENGINE_BRAKE_RATIO features.
    
    This is the core innovation of the pipeline. For EACH of the 128 feature sets,
    we dynamically calculate 6 composite features based on which proteins are present.
    These composites capture biological relationships and improve interpretability.
    
    CRITICAL: This function expects X_base to be ALREADY Z-SCORE NORMALIZED.
    Composite features are computed on Z-scores, not raw values.
    
    Parameters
    ----------
    X_base : np.ndarray, shape (n_cells, n_features)
        Log-transformed, Z-score normalized features for selected proteins
        MUST be Z-score normalized before calling this function!
    feature_set_config : dict
        Configuration dict from generate_all_feature_combinations()
        Must contain keys: 'gas', 'brakes', 'engine', 'valid_complexes', 'features'
    feature_names : list of str
        Names of features in X_base (must match feature_set_config['features'])
        
    Returns
    -------
    X_augmented : np.ndarray, shape (n_cells, n_features + 5)
        Original features plus [GAS, BRAKE, GAS_BRAKE_RATIO, ACTIVATED_ENGINE, ACT_ENGINE_BRAKE_RATIO]
    feature_names_augmented : list of str
        Names of all features in X_augmented
        
    Composite Feature Definitions (All Computed on Z-Scores)
    ---------------------------------------------------------
    1. **SUM_GAS**: Log10 of sum of activator proteins (CycD1, CycE)
       - Uses LogSumExp on Z-scores to approximate log(sum(abundance))
       - If < 2 GAS proteins: SUM_GAS = NaN (not a meaningful sum)
    
    2. **SUM_BRAKE**: Log10 of sum of inhibitor proteins (p21, p27)
       - Uses LogSumExp on Z-scores
       - If < 2 BRAKE proteins: SUM_BRAKE = NaN (not a meaningful sum)
    
    3. **SUM_ENGINES**: Log10 of sum of engine proteins (CDK2, CDK4, CDK6)
       - Uses LogSumExp on Z-scores
       - If < 2 ENGINE proteins: SUM_ENGINES = NaN
    
    4. **GAS_BRAKE_RATIO**: GAS - BRAKE (allows individual proteins or sums)
       - If ≥1 gas AND ≥1 brake: GAS_BRAKE_RATIO = (sum of gas proteins) - (sum of brake proteins)
       - If 0 gas OR 0 brake: GAS_BRAKE_RATIO = NaN
       - Captures the fundamental "Gas vs Brakes" concept at any scale
    
    5. **SUM_ACTIVATED_ENGINE**: Log10 of sum of biologically valid cyclin-CDK products
       - Complex formation (Mass Action): [Complex] ∝ [Cyclin] * [CDK]
       - In Log/Z-score space: Z(Complex) = Z(Cyclin) + Z(CDK)
       - Total Activated Engine = LogSumExp(Z(Complex_1), Z(Complex_2), ...)
       - If < 1 valid complex: SUM_ACTIVATED_ENGINE = NaN
       
       Examples:
       - {CycD1, CDK4} → Z(CycD1) + Z(CDK4) (1 complex)
       - {CycD1, CDK4, CDK6} → LogSumExp( Z(CycD1)+Z(CDK4), Z(CycD1)+Z(CDK6) )
    
    6. **ACT_ENGINE_BRAKE_RATIO**: SUM_ACTIVATED_ENGINE - BRAKE (allows individual brakes)
       - If ≥1 valid complex AND ≥1 brake: ACT_ENGINE_BRAKE_RATIO = SUM_ACTIVATED_ENGINE - (sum of brakes)
       - If 0 complexes OR 0 brakes: ACT_ENGINE_BRAKE_RATIO = NaN
    """
    n_cells = X_base.shape[0]
    n_base_features = X_base.shape[1]
    
    # Validate inputs
    if len(feature_names) != n_base_features:
        raise ValueError(
            f"feature_names length ({len(feature_names)}) must match "
            f"X_base columns ({n_base_features})"
        )
    
    if set(feature_names) != set(feature_set_config['features']):
        raise ValueError(
            "feature_names must match feature_set_config['features']. "
            f"Got {feature_names} vs {feature_set_config['features']}"
        )
    
    # Create mapping from protein name to column index
    protein_to_idx = {name: idx for idx, name in enumerate(feature_names)}
    
    # Initialize composite features
    feature_sum_gas = np.full(n_cells, np.nan)
    feature_sum_brake = np.full(n_cells, np.nan)
    feature_sum_engines = np.full(n_cells, np.nan)
    feature_gas_brake_ratio = np.full(n_cells, np.nan)
    feature_sum_activated_engine = np.full(n_cells, np.nan)
    feature_act_engine_brake_ratio = np.full(n_cells, np.nan)
    
    # Helper to apply logsumexp10 to columns (wraps module-level function)
    def logsumexp10_cols(arrays):
        """Apply logsumexp10 to a list of column arrays."""
        if not arrays:
            return np.full(n_cells, np.nan)
        # Stack columns: each array in arrays is shape (n_cells,)
        # We want to compute logsumexp10 across columns (axis=1)
        stacked = np.column_stack(arrays)  # shape: (n_cells, n_arrays)
        ln10 = np.log(10)
        return np.logaddexp.reduce(stacked * ln10, axis=1) / ln10

    # 1. Compute SUM_GAS: LogSumExp of Z-scores of activators (CycD1, CycE)
    if len(feature_set_config['gas']) >= 2:
        gas_cols = []
        for protein in feature_set_config['gas']:
            if protein in protein_to_idx:
                gas_cols.append(X_base[:, protein_to_idx[protein]])
        feature_sum_gas = logsumexp10_cols(gas_cols)
    
    # 2. Compute SUM_BRAKE: LogSumExp of Z-scores of inhibitors (p21, p27)
    if len(feature_set_config['brakes']) >= 2:
        brake_cols = []
        for protein in feature_set_config['brakes']:
            if protein in protein_to_idx:
                brake_cols.append(X_base[:, protein_to_idx[protein]])
        feature_sum_brake = logsumexp10_cols(brake_cols)
    
    # 3. Compute SUM_ENGINES: LogSumExp of Z-scores of engine proteins (CDK2, CDK4, CDK6)
    if len(feature_set_config['engine']) >= 2:
        engine_cols = []
        for protein in feature_set_config['engine']:
            if protein in protein_to_idx:
                engine_cols.append(X_base[:, protein_to_idx[protein]])
        feature_sum_engines = logsumexp10_cols(engine_cols)
    
    # 4. Compute GAS_BRAKE_RATIO: GAS - BRAKE (allows individual proteins or sums)
    if len(feature_set_config['gas']) >= 1 and len(feature_set_config['brakes']) >= 1:
        # Get Gas Component
        if len(feature_set_config['gas']) >= 2:
            gas_val = feature_sum_gas # Already computed
        else:
            # Single gas protein
            gas_val = X_base[:, protein_to_idx[feature_set_config['gas'][0]]]
            
        # Get Brake Component
        if len(feature_set_config['brakes']) >= 2:
            brake_val = feature_sum_brake # Already computed
        else:
            # Single brake protein
            brake_val = X_base[:, protein_to_idx[feature_set_config['brakes'][0]]]
            
        feature_gas_brake_ratio = gas_val - brake_val

    # 5. Compute SUM_ACTIVATED_ENGINE: LogSumExp of valid cyclin-CDK products
    if len(feature_set_config['valid_complexes']) >= 1:
        complex_terms = []
        
        for cyclin, cdk_partners in feature_set_config['valid_complexes']:
            if cyclin in protein_to_idx:
                cyclin_val = X_base[:, protein_to_idx[cyclin]]
                
                for cdk in cdk_partners:
                    if cdk in protein_to_idx:
                        cdk_val = X_base[:, protein_to_idx[cdk]]
                        # Complex = Cyclin * CDK (in linear) => Cyclin + CDK (in log/Z-score)
                        complex_terms.append(cyclin_val + cdk_val)
        
        if complex_terms:
            if len(complex_terms) == 1:
                feature_sum_activated_engine = complex_terms[0]
            else:
                feature_sum_activated_engine = logsumexp10_cols(complex_terms)
    
    # 6. Compute ACT_ENGINE_BRAKE_RATIO: SUM_ACTIVATED_ENGINE - BRAKE
    if len(feature_set_config['valid_complexes']) >= 1 and len(feature_set_config['brakes']) >= 1:
        # Get Engine Component (already computed or NaN)
        engine_val = feature_sum_activated_engine
        
        # Get Brake Component
        if len(feature_set_config['brakes']) >= 2:
            brake_val = feature_sum_brake
        else:
            brake_val = X_base[:, protein_to_idx[feature_set_config['brakes'][0]]]
            
        # Only compute if we have valid engine value
        feature_act_engine_brake_ratio = engine_val - brake_val
    
    # Stack all features together
    # Order: base features, then composites in this specific order
    X_augmented = np.column_stack([
        X_base,
        feature_sum_gas,
        feature_sum_brake,
        feature_sum_engines,
        feature_gas_brake_ratio,
        feature_sum_activated_engine,
        feature_act_engine_brake_ratio
    ])
    
    # Create augmented feature names (must match column order)
    feature_names_augmented = feature_names + [
        'Feature_SUM_GAS',
        'Feature_SUM_BRAKE',
        'Feature_SUM_ENGINES',
        'Feature_GAS_BRAKE_RATIO',
        'Feature_SUM_ACTIVATED_ENGINE',
