"""
Nested Cross-Validation Module for Elastic Net Analysis

This module implements nested cross-validation for unbiased performance estimation.

Key Features:
- Outer loop: RepeatedStratifiedKFold for performance estimation
- Inner loop: GridSearchCV for hyperparameter tuning
- Proper data flow to prevent leakage: raw → log → Z-score → composites → clean
- NaN handling: Remove features with all NaN, then samples with any NaN
- Error handling: Graceful warnings for edge cases, continues analysis

Critical Design:
- Purpose: Performance estimation ONLY (mean AUROC ± SE)
- No model saving: We'll redo hyperparameter search on full data later
- Prevents leakage: Test data uses train scaler statistics
"""

import numpy as np
import warnings
from typing import Dict, List, Tuple, Optional
from sklearn.model_selection import RepeatedStratifiedKFold, GridSearchCV, StratifiedKFold
from sklearn.linear_model import LogisticRegression
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import roc_auc_score

from .feature_engineering import compute_composite_features

def nested_cv_single_feature_set(
    adata,
    y: np.ndarray,
    feature_set_config: Dict,
    param_grid: Dict,
    outer_cv_splits: int = 10,
    outer_cv_repeats: int = 1,
    inner_cv: int = 5,
    random_state: int = 42,
    verbose: bool = True,
    return_fold_models: bool = False,
    use_composite_features: bool = True,
    feature_mode: Optional[str] = None
) -> Dict:
    """
    Perform nested cross-validation for a single feature set.
    
    This function provides unbiased performance estimates by using nested CV:
    - Outer loop: Splits data for performance estimation
    - Inner loop: Tunes hyperparameters on training data only
    
    Parameters
    ----------
    adata : AnnData
        Single-cell data object with RAW protein intensities in .X
        Function will extract only the proteins needed for this feature set
    y : np.ndarray, shape (n_cells,)
        Binary labels (0 = non-proliferative, 1 = proliferative)
    feature_set_config : dict
        Configuration from generate_all_feature_combinations()
        Must contain 'features' key with list of protein names
    param_grid : dict
        Hyperparameters to search for Elastic Net:
        Default: {'l1_ratio': [0.1, 0.3, 0.5, 0.7, 0.9], 'C': [0.1, 1.0, 10.0, 100.0]}
        - l1_ratio: Mix of L1/L2 penalty (0=Ridge, 1=Lasso, 0.5=equal mix)
        - C: Inverse regularization strength (smaller = more regularization)
    outer_cv_splits : int
        Number of splits for RepeatedStratifiedKFold outer loop (default 10)
        This determines the train/test split size: test_size = 1/outer_cv_splits
        For example: outer_cv_splits=10 → 90% train / 10% test per fold
    outer_cv_repeats : int
        Number of repeats for RepeatedStratifiedKFold outer loop (default 1)
        Total outer folds = outer_cv_splits × outer_cv_repeats
    inner_cv : int
        Number of inner CV folds for GridSearchCV (default 5)
    random_state : int
        Random seed for reproducibility
    verbose : bool
        Print progress information (outer fold progress)
    return_fold_models : bool
        If True, return trained models from each fold (for coefficient stability checks)
        If False, discard models after evaluation (saves memory, default)
        WARNING: Setting to True increases memory usage significantly
    use_composite_features : bool
        Legacy parameter. If True, equivalent to feature_mode='mixed'.
        If False, equivalent to feature_mode='base'.
        Ignored if feature_mode is explicitly provided.
    feature_mode : str, optional
        Determines which features are used for the model:
        - 'base': Use only base protein features (no composites)
        - 'mixed': Use base proteins + composite features (default behavior)
        - 'composite': Use ONLY composite features (drop base proteins)
        If None, inferred from use_composite_features.
        
    Returns
    -------
    results : dict
        {
            'mean_auroc': float (NaN if all folds failed),
            'se_auroc': float,  # Standard error of AUROC across outer folds
            'fold_scores': list of float,  # AUROC for each outer fold
            'feature_set_config': dict,  # For reference
            'n_features': int,  # Average features after NaN removal
            'warnings': list of str,  # Any warnings from folds
            'fold_models': list of dict (only if return_fold_models=True)
                Each dict contains:
                {
                    'model': LogisticRegression,  # Trained model from this fold
                    'scaler': StandardScaler,  # Fitted scaler from this fold
                    'feature_names': list of str,  # Features used (after NaN removal)
                    'best_params': dict,  # Hyperparameters selected by inner CV
                    'fold_idx': int  # Fold number
                }
        }
        
    Notes
    -----
    Critical for preventing data leakage:
    1. Raw data split first (outer fold)
    2. Log transform applied to train and test separately
    3. StandardScaler fit ONLY on train log data
    4. Test log data transformed using TRAIN scaler statistics
    5. Composites computed on Z-scored data (separately for train/test)
    6. NaN removal done after composites (columns, then rows)
    7. Inner GridSearchCV sees only clean train data
    
    This function is for PERFORMANCE ESTIMATION only. After finding the best
    feature set via 1-SE rule, we'll redo hyperparameter search on 100% of data.
    """
    # Sub-step 3.2.1: Configure feature mode and extract proteins
    # PURPOSE: Set feature engineering mode and extract protein subset from data
    # INPUTS: Feature set configuration, Feature mode setting, AnnData object
    # OUTPUTS: Configured feature_mode, Protein list, Raw protein matrix
    
    # Backwards compatibility logic
    if feature_mode is None:
        feature_mode = 'mixed' if use_composite_features else 'base'
    
    # Extract protein names for this feature set
    protein_list = feature_set_config['features']
    
    if not protein_list:
        # Empty feature set - return NaN
        warnings.warn("Empty feature set provided. Returning NaN AUROC.")
        return {
            'mean_auroc': np.nan,
            'se_auroc': np.nan,
            'fold_scores': [],
            'feature_set_config': feature_set_config,
            'n_features': 0,
            'warnings': ['Empty feature set']
        }
    
    # Sub-step 3.2.2: Extract protein subset from full dataset
    # PURPOSE: Select only the proteins needed for this feature set evaluation
    # INPUTS: AnnData object with all proteins, Selected protein names
    # OUTPUTS: Raw protein intensity matrix for selected proteins only
    
    # Extract raw data for selected proteins
    protein_indices = [list(adata.var_names).index(p) for p in protein_list]
    if hasattr(adata.X, 'toarray'):
        X_raw = adata.X[:, protein_indices].toarray()
    else:
        X_raw = adata.X[:, protein_indices]
    
    # Calculate test_size from outer_cv_splits
    test_size = 1.0 / outer_cv_splits
    
    if verbose:
        train_pct = (1 - test_size) * 100
        test_pct = test_size * 100
        print(f"\n{'='*80}")
        print(f"Nested CV: {feature_set_config['name']}")
        print(f"  Features: {protein_list}")
        print(f"  Outer CV: {outer_cv_splits} splits × {outer_cv_repeats} repeats = {outer_cv_splits * outer_cv_repeats} folds")
        print(f"  Train/Test: {train_pct:.0f}% / {test_pct:.0f}% per fold")
        print(f"  Inner CV: {inner_cv} folds")
        print(f"  Param grid: {len(param_grid.get('l1_ratio', [1]))} l1_ratios × {len(param_grid.get('C', [1]))} C values = {len(param_grid.get('l1_ratio', [1])) * len(param_grid.get('C', [1]))} combinations")
        print(f"{'='*80}")
    
    # Sub-step 3.2.3: Initialize cross-validation splitter
    # PURPOSE: Create outer CV strategy for performance estimation
    # INPUTS: CV configuration (n_splits, n_repeats, random_state)
    # OUTPUTS: RepeatedStratifiedKFold splitter object
    
    # Initialize outer CV
    outer_cv = RepeatedStratifiedKFold(
        n_splits=outer_cv_splits,
        n_repeats=outer_cv_repeats,
        random_state=random_state
    )
    
    # Sub-step 3.2.4: Initialize result containers
    # PURPOSE: Prepare storage for fold-level results and statistics
    # INPUTS: Configuration flags (return_fold_models)
    # OUTPUTS: Empty lists for scores, warnings, parameters, coefficients
    
    # Store results from each fold
    fold_scores = []
    fold_warnings = []
    n_features_per_fold = []
    fold_best_params = []
    fold_coefficients = []
    fold_feature_names = []
    fold_models = [] if return_fold_models else None
    
    # Sub-step 3.2.5: Execute outer CV loop
    # PURPOSE: Evaluate model on each train/test split for unbiased performance estimate
    # INPUTS: Data splits from outer_cv, Hyperparameter grid, Inner CV settings
    # OUTPUTS: AUROC score, best parameters, and statistics for each fold
    # CRITICAL: Each fold trains a separate model - this is the most time-intensive operation
    # FUNCTION: _outer_cv_fold
    
    # Run outer CV loop
    for fold_idx, (train_idx, test_idx) in enumerate(outer_cv.split(X_raw, y)):
        X_raw_train = X_raw[train_idx]
        X_raw_test = X_raw[test_idx]
        y_train = y[train_idx]
        y_test = y[test_idx]
        
        # Process this fold
        fold_result = _outer_cv_fold(
            X_raw_train, X_raw_test, y_train, y_test,
            feature_set_config, protein_list, param_grid,
            inner_cv, fold_idx, verbose, return_fold_models,
            feature_mode=feature_mode
        )
        
        fold_scores.append(fold_result['auroc'])
        n_features_per_fold.append(fold_result['n_features'])
        
        # Store best_params (None if fold failed)
        fold_best_params.append(fold_result.get('best_params', None))
        
        # Store coefficients and feature names (None if fold failed)
        fold_coefficients.append(fold_result.get('coefficients', None))
        fold_feature_names.append(fold_result.get('feature_names', None))
        
        if fold_result['warning']:
            fold_warnings.append(f"Fold {fold_idx+1}: {fold_result['warning']}")
        
        # Store model info if requested
        if return_fold_models and not np.isnan(fold_result['auroc']):
            fold_models.append({
                'model': fold_result['model'],
                'scaler': fold_result['scaler'],
                'feature_names': fold_result['feature_names'],
                'best_params': fold_result['best_params'],
                'fold_idx': fold_idx
            })
    
    # Sub-step 3.2.6: Aggregate fold results
    # PURPOSE: Compute summary statistics from all fold evaluations
    # INPUTS: Fold-level AUROC scores, Feature counts, Warnings
    # OUTPUTS: mean_auroc, se_auroc, avg_n_features, coefficient_stats
    
    # Compute statistics (ignore NaN values from failed folds)
    valid_scores = [s for s in fold_scores if not np.isnan(s)]
    
    if not valid_scores:
        mean_auroc = np.nan
        se_auroc = np.nan
        avg_n_features = 0
        
        # Check if this is a "No features" case (expected for some composite models)
        # Suppress warning if all folds failed due to lack of features
        is_no_features_case = fold_warnings and all("No features available" in w for w in fold_warnings)
        
        if not is_no_features_case:
            warnings.warn(f"All folds failed for feature set {feature_set_config['name']}")
    else:
        mean_auroc = np.mean(valid_scores)
        se_auroc = np.std(valid_scores) / np.sqrt(len(valid_scores))  # Standard error
        avg_n_features = int(np.mean([n for n in n_features_per_fold if n > 0]))
    
    # Aggregate coefficient statistics across folds
    coefficient_stats = _aggregate_coefficient_stats(fold_coefficients, fold_feature_names)
    
    if verbose:
        print(f"\n{'='*80}")
        print(f"Results: {feature_set_config['name']}")
        print(f"  Mean AUROC: {mean_auroc:.4f} ± {se_auroc:.4f}")
        print(f"  Valid folds: {len(valid_scores)}/{len(fold_scores)}")
        print(f"  Avg features: {avg_n_features}")
        if fold_warnings:
            print(f"  Warnings: {len(fold_warnings)}")
            for w in fold_warnings[:3]:  # Show first 3
                print(f"    - {w}")
        print(f"{'='*80}\n")
    
    results = {
        'mean_auroc': mean_auroc,
        'se_auroc': se_auroc,
        'fold_scores': fold_scores,
        'fold_best_params': fold_best_params,
        'feature_set_config': feature_set_config,
        'n_features': avg_n_features,
        'coefficient_stats': coefficient_stats,
        'warnings': fold_warnings
    }
    
    # Add fold models if requested
    if return_fold_models:
        results['fold_models'] = fold_models
    
    return results




# WORKFLOW_EXCLUDE: evaluation_overview, evaluation_detailed
def _outer_cv_fold(
    X_raw_train: np.ndarray,
    X_raw_test: np.ndarray,
    y_train: np.ndarray,
    y_test: np.ndarray,
    feature_set_config: Dict,
    protein_list: List[str],
    param_grid: Dict,
    inner_cv: int,
    fold_idx: int,
    verbose: bool,
    return_fold_models: bool = False,
    use_composite_features: bool = True,
    feature_mode: Optional[str] = None
) -> Dict:
    """
    Process one outer CV fold.
    
    This function encapsulates all logic for a single outer fold,
    keeping the main nested_cv function clean and readable.
    
    Parameters
    ----------
    X_raw_train : np.ndarray
        Raw training data for this fold
    X_raw_test : np.ndarray
        Raw test data for this fold
    y_train, y_test : np.ndarray
        Labels
    feature_set_config : dict
        Feature set configuration
    protein_list : list of str
        Names of proteins in X_raw
    param_grid : dict
        Hyperparameter grid for GridSearchCV
    inner_cv : int
        Number of inner CV folds
    fold_idx : int
        Current fold number (for logging)
    verbose : bool
        Print progress
    return_fold_models : bool
        If True, include trained model, scaler, and feature names in return dict
    use_composite_features : bool
        Legacy parameter. Ignored if feature_mode is provided.
    feature_mode : str, optional
        'base', 'mixed', or 'composite'. If None, inferred from use_composite_features.
    
    Returns
    -------
    fold_result : dict
        {
            'auroc': float (NaN if fold failed),
            'n_train_samples': int (after NaN removal),
            'n_test_samples': int (after NaN removal),
            'n_features': int (after NaN removal),
            'best_params': dict (from GridSearchCV) or None,
            'warning': str or None (if any issues occurred),
            'model': LogisticRegression (only if return_fold_models=True),
            'scaler': StandardScaler (only if return_fold_models=True),
            'feature_names': list of str (only if return_fold_models=True)
        }
    """
    # Sub-step 3.2.5.1: Configure fold parameters
    # PURPOSE: Set feature mode for this fold
    # INPUTS: Feature mode setting
    # OUTPUTS: Configured feature_mode
    
    # Backwards compatibility logic
    if feature_mode is None:
        feature_mode = 'mixed' if use_composite_features else 'base'
        
    warning = None
    
    try:
        # Sub-step 3.2.5.2: Prepare training features
        # PURPOSE: Transform raw training data through full preprocessing pipeline
        # INPUTS: Raw training data, Feature set config, Protein names
        # OUTPUTS: Clean training features, Feature names, Fitted scaler, Valid row mask
        # FUNCTION: _prepare_features
        
        # Step 1: Prepare train data (fit scaler)
        X_train_clean, feature_names_clean, scaler, train_valid = _prepare_features(
            X_raw_train, feature_set_config, protein_list, scaler=None,
            feature_mode=feature_mode
        )
        
        # Check if we have any features left
        if len(feature_names_clean) == 0:
            warning = "No features available (skipped)"
            return {
                'auroc': np.nan,
                'n_train_samples': X_train_clean.shape[0],
                'n_test_samples': 0,
                'n_features': 0,
                'best_params': None,
                'warning': warning
            }
            
        y_train_clean = y_train[train_valid]
        
        # Sub-step 3.2.5.3: Prepare test features
        # PURPOSE: Transform raw test data using training scaler (prevents leakage)
        # INPUTS: Raw test data, Fitted scaler from training, Feature set config
        # OUTPUTS: Clean test features matching training feature space
        # FUNCTION: _prepare_features
        
        # Step 2: Prepare test data (use train scaler - prevents leakage!)
        X_test_clean, _, _, test_valid = _prepare_features(
            X_raw_test, feature_set_config, protein_list, scaler=scaler,
            feature_mode=feature_mode
        )
        y_test_clean = y_test[test_valid]
        
        if verbose:
            print(f"  Fold {fold_idx+1}: Train={X_train_clean.shape}, Test={X_test_clean.shape}, Features={len(feature_names_clean)}")
        
        # Sub-step 3.2.5.4: Validate data quality
        # PURPOSE: Check for edge cases that would prevent model training
        # INPUTS: Cleaned training and test data, Class labels
        # OUTPUTS: Validation status (returns early with warning if invalid)
        
        # Step 3: Check for edge cases
        if X_train_clean.shape[0] < 10:
            warning = f"< 10 training samples after NaN removal ({X_train_clean.shape[0]})"
            return {
                'auroc': np.nan,
                'n_train_samples': X_train_clean.shape[0],
                'n_test_samples': X_test_clean.shape[0],
                'n_features': len(feature_names_clean),
                'best_params': None,
                'warning': warning
            }
        
        if X_test_clean.shape[0] < 1:
            warning = f"No test samples after NaN removal"
            return {
                'auroc': np.nan,
                'n_train_samples': X_train_clean.shape[0],
                'n_test_samples': 0,
                'n_features': len(feature_names_clean),
                'best_params': None,
                'warning': warning
            }
        
        # Check class balance
        train_classes = np.unique(y_train_clean)
        test_classes = np.unique(y_test_clean)
        
        if len(train_classes) < 2:
            warning = f"Only one class in training data: {train_classes}"
            return {
                'auroc': np.nan,
                'n_train_samples': X_train_clean.shape[0],
                'n_test_samples': X_test_clean.shape[0],
                'n_features': len(feature_names_clean),
                'best_params': None,
                'warning': warning
            }
        
        if len(test_classes) < 2:
            warning = f"Only one class in test data: {test_classes}"
            return {
                'auroc': np.nan,
                'n_train_samples': X_train_clean.shape[0],
                'n_test_samples': X_test_clean.shape[0],
                'n_features': len(feature_names_clean),
                'best_params': None,
                'warning': warning
            }
        
        # Sub-step 3.2.5.5: Run inner grid search
        # PURPOSE: Find optimal hyperparameters on training data only
        # INPUTS: Clean training data, Hyperparameter grid, Inner CV folds
        # OUTPUTS: Best hyperparameters, Fitted model
        # FUNCTION: GridSearchCV
        
        # Step 4: Run inner GridSearchCV
        model = LogisticRegression(
            penalty='elasticnet',
            solver='saga',
            class_weight='balanced',
            max_iter=1000,
            random_state=42
        )
        
        grid_search = GridSearchCV(
            model,
            param_grid,
            cv=StratifiedKFold(n_splits=inner_cv, shuffle=True, random_state=42),
            scoring='roc_auc',
            n_jobs=1,  # Use 1 job to avoid nested parallelism issues
            error_score='raise'
        )
        
        grid_search.fit(X_train_clean, y_train_clean)
        
        # Sub-step 3.2.5.6: Evaluate on test set
        # PURPOSE: Compute unbiased performance on held-out test data
        # INPUTS: Trained model with optimal hyperparameters, Clean test data
        # OUTPUTS: Test set AUROC, Model coefficients
        
        # Step 5: Evaluate on test data
        y_test_proba = grid_search.predict_proba(X_test_clean)[:, 1]
        auroc = roc_auc_score(y_test_clean, y_test_proba)
        
        # Extract coefficients from best model
        best_model = grid_search.best_estimator_
        coefficients = best_model.coef_[0].tolist()  # Convert to list for JSON serialization
        
        result = {
            'auroc': auroc,
            'n_train_samples': X_train_clean.shape[0],
            'n_test_samples': X_test_clean.shape[0],
            'n_features': len(feature_names_clean),
            'best_params': grid_search.best_params_,
            'coefficients': coefficients,
            'feature_names': feature_names_clean,
            'warning': None
        }
        
        # Add model info if requested
        if return_fold_models:
            result['model'] = grid_search.best_estimator_
            result['scaler'] = scaler
            result['feature_names'] = feature_names_clean
        
        return result
        
    except Exception as e:
        warning = f"Exception in fold: {str(e)}"
        warnings.warn(warning)
        return {
            'auroc': np.nan,
            'n_train_samples': 0,
            'n_test_samples': 0,
            'n_features': 0,
            'best_params': None,
            'warning': warning
        }


# WORKFLOW_EXCLUDE: evaluation_overview, evaluation_detailed
def _prepare_features(
    X_raw: np.ndarray,
    feature_set_config: Dict,
    protein_list: List[str],
    scaler: Optional[StandardScaler] = None,
    pseudocount: float = 1e-100,
    use_composite_features: bool = True,
    feature_mode: Optional[str] = None
) -> Tuple[np.ndarray, List[str], StandardScaler, np.ndarray]:
    """
    Transform raw data → log → Z-score → composites (optional) → clean (no NaN).
    
    This modular function handles all feature engineering and preprocessing
    for both train and test data, preventing code duplication.
    
    Parameters
    ----------
    X_raw : np.ndarray, shape (n_cells, n_proteins)
        Raw protein intensities for selected proteins (should already be filtered to remove zeros)
    feature_set_config : dict
        Configuration from generate_all_feature_combinations()
    protein_list : list of str
        Names of proteins (columns in X_raw)
    scaler : StandardScaler or None
        If None: FIT mode - create and fit new scaler (for TRAIN data)
        If provided: TRANSFORM mode - use this scaler (for TEST data - prevents leakage!)
    pseudocount : float
        Minimal safety value added before log transform (default: 1e-100)
        Data should already be filtered; this is just protection
    use_composite_features : bool
        Legacy parameter. Ignored if feature_mode is provided.
    feature_mode : str, optional
        'base', 'mixed', or 'composite'. If None, inferred from use_composite_features.
    
    Returns
    -------
    X_clean : np.ndarray, shape (n_valid_cells, n_valid_features)
        Clean feature matrix with no NaN values
    feature_names_clean : list of str
        Names of features in X_clean (after removing NaN columns)
    scaler : StandardScaler
        The scaler used (newly fitted or the provided one)
    valid_row_indices : np.ndarray (boolean)
        Boolean mask indicating which rows (cells) survived NaN removal
    """
    # Sub-step 3.2.5.2.1: Configure feature mode
    # PURPOSE: Set feature engineering strategy
    # INPUTS: Feature mode setting
    # OUTPUTS: Configured feature_mode
    
    # Backwards compatibility logic
    if feature_mode is None:
        feature_mode = 'mixed' if use_composite_features else 'base'
        
    # Sub-step 3.2.5.2.2: Apply log transformation
    # PURPOSE: Log-transform raw protein intensities for normality
    # INPUTS: Raw protein matrix, Pseudocount for numerical stability
    # OUTPUTS: Log10-transformed protein matrix
    
    # Step 1: Log transform with minimal pseudocount for safety
    with warnings.catch_warnings():
        warnings.filterwarnings('ignore', category=RuntimeWarning)
        X_log = np.log10(X_raw + pseudocount)
    
    # Sub-step 3.2.5.2.3: Apply Z-score normalization
    # PURPOSE: Standardize features to zero mean and unit variance
    # INPUTS: Log-transformed data, Optional scaler (fit mode vs transform mode)
    # OUTPUTS: Z-scored protein matrix, Fitted or provided scaler
    # FUNCTION: StandardScaler
    
    # Step 2: Z-score
    if scaler is None:
        # FIT mode: Create and fit new scaler (TRAIN data)
        scaler = StandardScaler()
        X_zscore = scaler.fit_transform(X_log)
    else:
        # TRANSFORM mode: Use provided scaler (TEST data - prevents leakage!)
        X_zscore = scaler.transform(X_log)
    
    # Sub-step 3.2.5.2.4: Generate composite features
    # PURPOSE: Add engineered features based on biological relationships
    # INPUTS: Z-scored protein matrix, Feature set config, Feature mode
    # OUTPUTS: Augmented feature matrix with base proteins and/or composites
    # FUNCTION: compute_composite_features
    
    # Step 3: Compute composite features (optional)
    if feature_mode == 'mixed':
        # Use base proteins + composites
        X_augmented, feature_names = compute_composite_features(
            X_zscore, feature_set_config, protein_list
        )
    elif feature_mode == 'composite':
        # Use ONLY composites (drop base proteins)
        X_full, names_full = compute_composite_features(
            X_zscore, feature_set_config, protein_list
        )
        
        # Identify composite columns (those not in protein_list)
        n_base = len(protein_list)
        X_augmented = X_full[:, n_base:]
        feature_names = names_full[n_base:]
        
    else: # feature_mode == 'base'
        # Use only base proteins (no composites)
        X_augmented = X_zscore
        feature_names = protein_list
    
    # Sub-step 3.2.5.2.5: Remove NaN features
    # PURPOSE: Drop features that are entirely NaN (invalid composites)
    # INPUTS: Augmented feature matrix
    # OUTPUTS: Matrix with NaN columns removed, Updated feature names
    # FUNCTION: _remove_nan_features
    
    # Step 4: Remove NaN columns (features with all NaN)
    X_no_nan_cols, feature_names_clean, _ = _remove_nan_features(
        X_augmented, feature_names
    )
    
    # Sub-step 3.2.5.2.6: Remove NaN samples
    # PURPOSE: Drop samples with any remaining NaN values
    # INPUTS: Feature matrix without NaN columns
    # OUTPUTS: Fully clean feature matrix, Valid row indices
    # FUNCTION: _remove_nan_samples
    
    # Step 5: Remove NaN rows (samples with any NaN)
    X_clean, valid_row_indices, _ = _remove_nan_samples(X_no_nan_cols)
    
    return X_clean, feature_names_clean, scaler, valid_row_indices


def _remove_nan_features(
    X: np.ndarray,
    feature_names: List[str]
) -> Tuple[np.ndarray, List[str], List[str]]:
    """
    Remove columns (features) that are ALL NaN.
    
    Composite features may be all NaN when insufficient components:
    - SUM_GAS: NaN if < 2 gas proteins (CycD1, CycE)
    - SUM_BRAKE: NaN if < 2 brake proteins (p21, p27)
    - GAS_BRAKE_RATIO: NaN if either SUM_GAS or SUM_BRAKE is NaN
    - SUM_ACTIVATED_ENGINE: NaN if < 2 valid cyclin-CDK complexes
    - ACT_ENGINE_BRAKE_RATIO: NaN if either SUM_ACTIVATED_ENGINE or SUM_BRAKE is NaN
    
    Parameters
    ----------
    X : np.ndarray, shape (n_cells, n_features)
        Feature matrix (may contain columns with all NaN)
    feature_names : list of str
        Names of features (columns in X)
    
    Returns
    -------
    X_clean : np.ndarray, shape (n_cells, n_valid_features)
        Features with NaN columns removed
    feature_names_clean : list of str
        Updated feature names (only non-NaN columns)
    removed_features : list of str
        Names of features that were removed (for logging)
    """
    # Find columns that are ALL NaN
    all_nan_mask = np.all(np.isnan(X), axis=0)
    
    # Keep columns that are NOT all NaN
    valid_cols_mask = ~all_nan_mask
    
    X_clean = X[:, valid_cols_mask]
    feature_names_clean = [name for name, is_valid in zip(feature_names, valid_cols_mask) if is_valid]
    removed_features = [name for name, is_nan in zip(feature_names, all_nan_mask) if is_nan]
    
    return X_clean, feature_names_clean, removed_features


def _remove_nan_samples(
    X: np.ndarray
) -> Tuple[np.ndarray, np.ndarray, int]:
    """
    Remove rows (samples) that have ANY NaN values.
    
    After removing NaN columns, some rows may still have NaN values
    (e.g., if a cell has NaN in a base feature due to measurement error,
    or if composite feature logic creates NaN for specific cells).
    
    Parameters
    ----------
    X : np.ndarray, shape (n_cells, n_features)
        Feature matrix (may contain NaN in some rows)
    
    Returns
    -------
    X_clean : np.ndarray, shape (n_valid_cells, n_features)
        Samples with NaN removed
    valid_row_indices : np.ndarray (boolean), shape (n_cells,)
        Boolean mask indicating which rows survived
    n_removed : int
        Number of samples removed (for logging)
    """
    # Find rows with ANY NaN
    any_nan_mask = np.any(np.isnan(X), axis=1)
    
    # Keep rows that have NO NaN
    valid_rows_mask = ~any_nan_mask
    
    X_clean = X[valid_rows_mask, :]
    n_removed = np.sum(any_nan_mask)
    
    return X_clean, valid_rows_mask, n_removed


def _aggregate_coefficient_stats(
    fold_coefficients: List[Optional[List[float]]],
    fold_feature_names: List[Optional[List[str]]]
) -> Dict:
    """
    Aggregate coefficient statistics across CV folds.
    
    This function handles variable feature sets across folds (due to NaN removal)
    and computes mean, std, SE, and sign flip detection for each feature.
    
    Parameters
    ----------
    fold_coefficients : list of list of float (or None)
        Coefficients from each fold. None if fold failed.
        Each inner list has coefficients matching fold_feature_names[i]
    fold_feature_names : list of list of str (or None)
        Feature names from each fold. None if fold failed.
        
    Returns
    -------
    coefficient_stats : dict
        {
            'feature_names': list of str,  # All unique features (sorted)
            'mean_coef': list of float,  # Mean coefficient per feature (across valid folds)
            'std_coef': list of float,  # Std deviation per feature
            'se_coef': list of float,  # Standard error per feature
            'sign_flip_detected': list of bool,  # True if sign changes across folds
            'sign_consistency': list of float,  # Ratio of dominant sign (1.0 = all same sign)
            'n_folds_positive': list of int,  # Number of folds with positive coefficient
            'n_folds_negative': list of int,  # Number of folds with negative coefficient
            'n_folds_present': list of int,  # Number of folds where feature was present
            'fold_coefficients': list of list,  # Raw coefficients per fold (aligned by feature)
        }
    """
    from collections import defaultdict
    
    # Collect coefficients by feature name
    feature_coef_map = defaultdict(list)
    
    for fold_coefs, fold_names in zip(fold_coefficients, fold_feature_names):
        if fold_coefs is None or fold_names is None:
            continue  # Skip failed folds
        
        # Map coefficients to feature names for this fold
        for coef, name in zip(fold_coefs, fold_names):
            feature_coef_map[name].append(coef)
    
    # Get all unique features (sorted for consistency)
    all_features = sorted(feature_coef_map.keys())
    
    if not all_features:
        # No valid folds - return empty stats
        return {
            'feature_names': [],
            'mean_coef': [],
            'std_coef': [],
            'se_coef': [],
            'sign_flip_detected': [],
            'sign_consistency': [],
            'n_folds_positive': [],
            'n_folds_negative': [],
            'n_folds_present': [],
            'fold_coefficients': []
        }
    
    # Calculate statistics for each feature
    mean_coefs = []
    std_coefs = []
    se_coefs = []
    sign_flips = []
    sign_consistencies = []
    n_pos_list = []
    n_neg_list = []
    n_present_list = []
    fold_coef_matrix = []
    
    for feature in all_features:
        coefs = np.array(feature_coef_map[feature])
        n_folds = len(coefs)
        
        # Basic statistics
        mean_coef = np.mean(coefs)
        std_coef = np.std(coefs, ddof=1) if n_folds > 1 else 0.0
        se_coef = std_coef / np.sqrt(n_folds) if n_folds > 1 else 0.0
        
        # Sign flip detection
        n_positive = np.sum(coefs > 0)
        n_negative = np.sum(coefs < 0)
        n_zero = np.sum(coefs == 0)
        
        # Sign flip occurs if both positive and negative coefficients exist
        has_sign_flip = (n_positive > 0) and (n_negative > 0)
        
        # Consistency: fraction of folds with the dominant sign
        dominant_count = max(n_positive, n_negative)
        consistency = dominant_count / n_folds if n_folds > 0 else 0.0
        
        # Store results
        mean_coefs.append(float(mean_coef))
        std_coefs.append(float(std_coef))
        se_coefs.append(float(se_coef))
        sign_flips.append(bool(has_sign_flip))
        sign_consistencies.append(float(consistency))
        n_pos_list.append(int(n_positive))
        n_neg_list.append(int(n_negative))
        n_present_list.append(int(n_folds))
        fold_coef_matrix.append(coefs.tolist())
    
    return {
        'feature_names': all_features,
        'mean_coef': mean_coefs,
        'std_coef': std_coefs,
        'se_coef': se_coefs,
        'sign_flip_detected': sign_flips,
        'sign_consistency': sign_consistencies,
        'n_folds_positive': n_pos_list,
        'n_folds_negative': n_neg_list,
        'n_folds_present': n_present_list,
        'fold_coefficients': fold_coef_matrix
    }
